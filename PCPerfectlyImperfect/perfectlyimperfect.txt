    function shuffle(array) {
        let currentIndex = array.length,  randomIndex;
        while (currentIndex != 0) {
            randomIndex = Math.floor(Math.random() * currentIndex);
            currentIndex--;
            [array[currentIndex], array[randomIndex]] = [array[randomIndex], array[currentIndex]];
        }
        return array;
    }

    function resize() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        player.x = canvas.width / 2 - player.w / 2;

    function start() {
        document.getElementById('start-screen').style.display = 'none';
        state.active = true;
        startTimer();

    function startTimer() {
        const timerInt = setInterval(() => {
            if (!state.active) {
                clearInterval(timerInt);
                return;
            }
            if (state.paused) return;
            state.timeLeft--;
            const m = Math.floor(state.timeLeft / 60).toString().padStart(2, '0');
            const s = (state.timeLeft % 60).toString().padStart(2, '0');
            document.getElementById('timer-display').innerText = `${m}:${s}`;
            if (state.timeLeft <= 0) gameOver("TIME EXPIRED");
        }, 1000);
    }

    function fire() {
        if (!state.active || state.paused || state.isFalling) return;
        bullets.push({ x: player.x + player.w / 2 - 2, y: player.y, w: 4, h: 25 });
    }

    function spawn() {
        if (!state.active || state.paused || state.isFalling) return;
        const now = Date.now();
        if (now - state.lastSpawn > (state.inTunnel ? 2200 : 1800)) {
            enemies.push({
                x: Math.random() * (canvas.width - 50),
                y: -60, w: 50, h: 50,
                speed: state.inTunnel ? 1.5 : (2 + Math.random() * 1.5),
                type: 'thought'
            });
            state.lastSpawn = now;
        }
        if (!state.inTunnel && now - state.lastStar > 8000) {
            stars.push({
                x: Math.random() * (canvas.width - 40),
                y: -50, w: 40, h: 40, speed: 3.5, type: 'star'
            });
            state.lastStar = now;
        }
    }

    function update() {
        if (!state.active || state.paused) return;

        // Apply constant gravity when not on ground
        if (state.isFalling) {
            player.vy += 0.5;
            player.y += player.vy;
            // Detect bottom of screen transition
            if (player.y > canvas.height + 100) {
                enterTunnel();
            }
            return;
        }

        // Movement
        if (player.movingLeft) player.x -= player.speed;
        if (player.movingRight) player.x += player.speed;
        player.x = Math.max(0, Math.min(canvas.width - player.w, player.x));

        // Physics for all states
        player.vy += 0.5; // Constant gravity
        player.y += player.vy;

        // Ground detection
        const groundY = state.inTunnel ? (canvas.height - 120) : (canvas.height - 180 + player.h); 
        if (player.y + player.h > groundY) {
            player.y = groundY - player.h;
            player.vy = 0;
        }

        bullets.forEach((b, i) => {
            b.y -= 14;
            if (b.y < -30) bullets.splice(i, 1);
        });

        // Object Cleanup & Collisions
        enemies.forEach((e, i) => {
            e.y += e.speed;
            if (e.y > canvas.height) {
                enemies.splice(i, 1);
            }
            bullets.forEach((b, bi) => {
                if (rectIntersect(b, e)) {
                    bullets.splice(bi, 1);
                    enemies.splice(i, 1);
                    createExplosion(e.x + e.w/2, e.y + e.h/2, '#3b82f6');
                    triggerQuiz();
                }
            });
        });

        stars.forEach((s, i) => {
            s.y += s.speed;
            bullets.forEach((b, bi) => {
                if (rectIntersect(b, s)) {
                    bullets.splice(bi, 1);
                    stars.splice(i, 1);
                    createExplosion(s.x + s.w/2, s.y + s.h/2, '#fbbf24');
                    triggerFact();
                }
            });
        });

        particles.forEach((p, i) => {
            p.x += p.vx; p.y += p.vy; p.life -= 0.02;
            if (p.life <= 0) particles.splice(i, 1);
        });
    }

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);

        // Backgrounds
        if (state.isFalling) {
            ctx.fillStyle = '#0f172a';
            ctx.fillRect(player.x - 20, -5000, 100, 10000);
        } else if (state.inTunnel) {
            ctx.fillStyle = '#1e1b4b';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = '#020617'; 
            ctx.fillRect(0, canvas.height - 120, canvas.width, 120);
        }

        // Bullets
        ctx.fillStyle = '#60a5fa';
        bullets.forEach(b => ctx.fillRect(b.x, b.y, b.w, b.h));

        // Enemies
        enemies.forEach(e => {
            ctx.fillStyle = state.inTunnel ? '#ef4444' : '#3b82f6';
            ctx.beginPath();
            ctx.moveTo(e.x + e.w/2, e.y);
            ctx.lineTo(e.x + e.w, e.y + e.h/2);
            ctx.lineTo(e.x + e.w/2, e.y + e.h);
            ctx.lineTo(e.x, e.y + e.h/2);
            ctx.closePath();
            ctx.fill();
            ctx.strokeStyle = 'white';
            ctx.lineWidth = 2;
            ctx.stroke();
        });

        // Growth Stars
        stars.forEach(s => {
            ctx.fillStyle = '#fbbf24';
            ctx.beginPath();
            for(let i=0; i<5; i++) {
                ctx.lineTo(Math.cos((18+i*72)/180*Math.PI)*20+s.x+20, Math.sin((18+i*72)/180*Math.PI)*20+s.y+20);
                ctx.lineTo(Math.cos((54+i*72)/180*Math.PI)*10+s.x+20, Math.sin((54+i*72)/180*Math.PI)*10+s.y+20);
            }
            ctx.closePath(); ctx.fill();
        });

        // Effects
        particles.forEach(p => {
            ctx.globalAlpha = p.life;
            ctx.fillStyle = p.color;
            ctx.fillRect(p.x, p.y, 4, 4);
        });
        ctx.globalAlpha = 1;

        // Player Avatar
        ctx.save();
        ctx.translate(player.x + player.w/2, player.y + player.h/2);
        if (!player.facingRight) ctx.scale(-1, 1);
        
        if (state.inTunnel) {
            ctx.fillStyle = '#7c2d12';
            ctx.fillRect(-25, 10, 50, 20);
            ctx.fillStyle = '#ffdbac';
            ctx.fillRect(20, 5, 15, 15);
        } else {
            ctx.fillStyle = '#fff';
            ctx.fillRect(-20, -25, 40, 45);
            ctx.fillStyle = '#3b82f6';
            ctx.fillRect(-12, -35, 24, 15);
            ctx.fillStyle = '#ffdbac';
            ctx.fillRect(-10, -32, 20, 10);
        }
        ctx.restore();
    }

    function triggerQuiz() {
        state.paused = true;
        if (activeDeck.length === 0) activeDeck = shuffle([...MCQS]);
        const q = activeDeck[0];
        const modal = document.getElementById('modal-overlay');
        const feedback = document.getElementById('feedback-text');
        modal.style.display = 'flex';
        document.getElementById('quiz-content').classList.remove('hidden');
        document.getElementById('fact-content').classList.add('hidden');
        document.getElementById('question-text').innerText = q.q;
        feedback.innerText = "";
        
        const container = document.getElementById('options-container');
        container.innerHTML = '';

        const shuffledOptions = shuffle([...q.o]);

        shuffledOptions.forEach(opt => {
            const btn = document.createElement('button');
            btn.className = 'option-btn';
            btn.innerText = opt;
            btn.onclick = () => {
                const btns = container.querySelectorAll('button');
                btns.forEach(b => b.disabled = true);
                
                if (opt === q.a) {
                    btn.classList.add('correct');
                    state.score += 100;
                    document.getElementById('score-display').innerText = state.score.toString().padStart(4, '0');
                    createFireworks(canvas.width / 2, canvas.height / 2);
                    feedback.innerText = "Insight Gained! Returning to Growth Zone...";
                    setTimeout(() => {
                        activeDeck.shift();
                        if (activeDeck.length === 0) activeDeck = shuffle([...MCQS]);
                        if (state.inTunnel) exitTunnel();
                        resumeGame();
                    }, 1200);
                } else {
                    btn.classList.add('wrong');
                    feedback.innerText = "Mindset Shift Needed! Truth: " + q.a;
                    damage(); 

                    setTimeout(() => {
                        if (!state.active) return; 
                        const missed = activeDeck.shift();
                        activeDeck.splice(3, 0, missed);
                        if (activeDeck.length === 0) activeDeck = shuffle([...MCQS]);
                        
                        initTunnelFall();
                        resumeGame();
                    }, 2800);
                }
            };
            container.appendChild(btn);
        });
    }

    function triggerFact() {
        state.paused = true;
        const fact = FACTS[Math.floor(Math.random() * FACTS.length)];
        document.getElementById('modal-overlay').style.display = 'flex';
        document.getElementById('quiz-content').classList.add('hidden');
        document.getElementById('fact-content').classList.remove('hidden');
        document.getElementById('fact-text').innerText = fact;
        state.score += 50;
        document.getElementById('score-display').innerText = state.score.toString().padStart(4, '0');
    }

    function initTunnelFall() {
        state.isFalling = true;
        player.vy = 0;
    }

    function enterTunnel() {
        state.inTunnel = true;
        state.isFalling = false;
        document.getElementById('sector-label').innerText = "Tunnel of Doubt";
        document.getElementById('sector-label').style.color = "#ef4444";
        player.h = 30; 
        player.vy = 0;
        player.y = -100; // Start high to fall into view
        player.x = canvas.width / 2;
        enemies = [];
    }

    function exitTunnel() {
        state.inTunnel = false;
        document.getElementById('sector-label').innerText = "The Growth Zone";
        document.getElementById('sector-label').style.color = "#93c5fd";
        player.h = 60;
        player.y = -100;
        player.vy = 0;
        enemies = [];
    }

    function resumeGame() {
        document.getElementById('modal-overlay').style.display = 'none';
        state.paused = false;
    }

    function damage() {
        state.lives--;
        const hearts = '❤️'.repeat(Math.max(0, state.lives)) + '🖤'.repeat(5 - Math.max(0, state.lives));
        document.getElementById('lives-display').innerText = hearts;
        document.getElementById('game-container').classList.add('shake');
        setTimeout(() => document.getElementById('game-container').classList.remove('shake'), 300);
        if (state.lives <= 0) gameOver("RESILIENCE DEPLETED");
    }

    function createExplosion(x, y, color) {
        for (let i = 0; i < 15; i++) {
            particles.push({ x, y, vx: (Math.random()-0.5)*8, vy: (Math.random()-0.5)*8, life: 1, color });
        }
    }

    function createFireworks(x, y) {
        for (let i = 0; i < 50; i++) {
            particles.push({ x, y, vx: (Math.random()-0.5)*18, vy: (Math.random()-0.5)*18, life: 2.2, color: `hsl(${Math.random()*360}, 100%, 70%)` });
        }
    }

    function rectIntersect(r1, r2) {
        return r1.x < r2.x + r2.w && r1.x + r1.w > r2.x && r1.y < r2.y + r2.h && r1.y + r1.h > r2.y;
    }

    function loop() {
        spawn();
        update();
        draw();
        requestAnimationFrame(loop);
    }

    function gameOver(reason) {
        state.active = false;
        document.getElementById('end-screen').style.display = 'flex';
        document.getElementById('end-title').innerText = "Journey Ended";
        document.getElementById('end-reason').innerText = reason;
        document.getElementById('final-score').innerText = state.score.toString().padStart(4, '0');
    }
