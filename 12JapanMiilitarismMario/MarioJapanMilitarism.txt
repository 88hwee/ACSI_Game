        // --- Bulletproof safeRoundRect Helper to avoid browser-specific roundRect failures ---
        function safeRoundRect(c, x, y, w, h, r) {
            if (typeof r === 'undefined') {
                r = 0;
            }
            let tl = 0, tr = 0, br = 0, bl = 0;
            if (typeof r === 'number') {
                tl = tr = br = bl = r;
            } else if (Array.isArray(r)) {
                tl = r[0] || 0;
                tr = r[1] || 0;
                br = r[2] || 0;
                bl = r[3] || 0;
            } else {
                tl = r.tl || 0;
                tr = r.tr || 0;
                br = r.br || 0;
                bl = r.bl || 0;
            }
            c.beginPath();
            c.moveTo(x + tl, y);
            c.lineTo(x + w - tr, y);
            c.quadraticCurveTo(x + w, y, x + w, y + tr);
            c.lineTo(x + w, y + h - br);
            c.quadraticCurveTo(x + w, y + h, x + w - br, y + h);
            c.lineTo(x + bl, y + h);
            c.quadraticCurveTo(x, y + h, x, y + h - bl);
            c.lineTo(x, y + tl);
            c.quadraticCurveTo(x, y, x + tl, y);
            c.closePath();
        }

        // --- SHUFFLE HELPER ---
        function shuffleArray(arr) {
            for (let i = arr.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [arr[i], arr[j]] = [arr[j], arr[i]];
            }
            return arr;
        }

        // --- DYNAMIC SELECTION OF 35 DENSE QUESTIONS ---
        function selectChronologicalQuestions() {
            // Shuffle the complete pool
            let pool = [...masterQuestionPool];
            shuffleArray(pool);

            // Select 35 questions to spread across the dense run
            let selected = pool.slice(0, 35);

            // Sort them by Era so the student experiences a logical chronological timeline!
            selected.sort((a, b) => {
                let eraA = parseInt(a.era.match(/\d+/)?.[0] || "1");
                let eraB = parseInt(b.era.match(/\d+/)?.[0] || "1");
                return eraA - eraB;
            });

            // Shuffle choice layouts while keeping correct indices accurate
            return selected.map(q => {
                const choiceMap = q.choices.map((choiceText, index) => ({
                    text: choiceText,
                    isCorrect: index === q.correct
                }));

                shuffleArray(choiceMap);

                return {
                    ...q,
                    choices: choiceMap.map(c => c.text),
                    correct: choiceMap.findIndex(c => c.isCorrect)
                };
            });
        }

        // --- NINJA SPAWN AND ATTACK ---
        function spawnNinjaAttack() {
            ninjaAttackActive = true;
            try {
                sfx.playWrong();
            } catch (e) {}
            
            ninja = {
                x: player.x + 400,
                y: player.y - 10,
                vx: -6.5,
                vy: 0,
                state: 'spawning',
                animTimer: 0,
                shurikenThrown: false
            };

            triggerFlashMessage("NINJA INBOUND! 🥷", "bg-red-700/90");
        }

        function updateNinja() {
            if (!ninja) return;

            ninja.animTimer += 0.1;

            if (ninja.state === 'spawning') {
                ninja.x += ninja.vx;
                
                if (ninja.x - player.x < 180) {
                    ninja.state = 'throwing';
                    ninja.vx = 0;
                    ninja.animTimer = 0;
                }
            } else if (ninja.state === 'throwing') {
                if (ninja.animTimer > 3 && !ninja.shurikenThrown) {
                    ninja.shurikenThrown = true;
                    projectiles.push({
                        x: ninja.x - 10,
                        y: ninja.y + 15,
                        vx: -9,
                        vy: 0,
                        type: 'shuriken',
                        angle: 0,
                        radius: 12
                    });
                    
                    ninja.state = 'retreating';
                    ninja.vx = 8;
                }
            } else if (ninja.state === 'retreating') {
                ninja.x += ninja.vx;
                // Pink puff particles on retreat! (v cute)
                if (Math.random() < 0.25) {
                    floatingScores.push({
                        x: ninja.x,
                        y: ninja.y + 15,
                        text: "🌸",
                        timer: 0.6
                    });
                }

                if (ninja.x > player.x + 600) {
                    ninja = null;
                    ninjaAttackActive = false;
                }
            }
        }

                ctx.translate(Math.floor(p.x - viewX), Math.floor(p.y));

                if (p.type === 'ground') {
                    ctx.fillStyle = '#15803d'; // cute cartoon grass
                    ctx.fillRect(0, 0, p.width, 10);
                    
                    ctx.fillStyle = '#78350f'; // rich brown dirt
                    ctx.fillRect(0, 10, p.width, p.height - 10);
                } else if (p.type === 'bridge' || p.type === 'block') {
                    ctx.fillStyle = '#b91c1c'; // Vermilion structures
                    ctx.fillRect(0, 0, p.width, p.height);
                }

                ctx.restore();
            });

            drawToriiGate(80, 1, viewX);
            drawToriiGate(5150, 1.25, viewX);
        }

        function updateSakura() {
            particlePetals.forEach(p => {
                p.y += p.speedY;
                p.x += Math.sin(p.angle) * 0.4 + p.speedX;
                p.angle += 0.02;

                if (p.y > canvas.height) {
                    p.y = -10;
                    p.x = Math.random() * canvas.width;
                }
            });
        }

        function drawSakura() {
            particlePetals.forEach(p => {
                ctx.save();
                ctx.translate(Math.floor(p.x), Math.floor(p.y));
                ctx.rotate(p.angle);
                
                ctx.fillStyle = '#fbcfe8'; 
                ctx.beginPath();
                ctx.ellipse(0, 0, p.size, p.size * 1.4, 0, 0, Math.PI * 2);
                ctx.fill();
                
                ctx.restore();
            });
        }

        // Generic bounding-box helper
        function isColliding(r1, r2) {
            return r1.x < r2.x + r2.width &&
                   r1.x + r1.width > r2.x &&
                   r1.y < r2.y + r2.height &&
                   r1.y + r1.height > r2.y;
        }

        // --- BULLETPROOF SEPARATE-AXIS MOVEMENT AND COLLISION ---
        function updatePlayerPhysics() {
            if (player.isDying) {
                player.deathTimer += 1;
                player.vy += 0.4; 
                player.y += player.vy;
                
                if (player.y > canvas.height + 100) {
                    handleDeathRespawn();
                }
                return;
            }

            // Apply horizontal drag/friction
            if (player.grounded) {
                player.vx *= 0.85;
            } else {
                player.vx *= 0.95;
            }

            // Keyboard input acceleration
            if (keys.left) {
                player.vx = -player.speed;
                player.facing = 'left';
            }
            if (keys.right) {
                player.vx = player.speed;
                player.facing = 'right';
            }

            // Apply gravity constant
            player.vy += player.gravity;

            // 1. Resolve Horizontal axis movement & collisions
            player.x += player.vx;
            if (player.x < 0) {
                player.x = 0;
                player.vx = 0;
            }
            if (player.x > LEVEL_WIDTH - player.width) {
                player.x = LEVEL_WIDTH - player.width;
                player.vx = 0;
            }

            // Snap against vertical surfaces
            platforms.forEach(p => {
                if (isColliding(player, p)) {
                    if (player.vx > 0) {
                        player.x = p.x - player.width;
                    } else if (player.vx < 0) {
                        player.x = p.x + p.width;
                    }
                    player.vx = 0;
                }
            });

            // 2. Resolve Vertical axis movement & collisions
            player.grounded = false;
            player.y += player.vy;

            platforms.forEach(p => {
                if (isColliding(player, p)) {
                    if (player.vy > 0) { // landing on top of standard platforms
                        player.y = p.y - player.height;
                        player.vy = 0;
                        player.grounded = true;
                    } else if (player.vy < 0) { // hitting ceilings
                        player.y = p.y + p.height;
                        player.vy = 0;
                    }
                }
            });

            // Trigger victory flag when crossing the final Torii gate
            if (player.x > 5150) {
                concludeGame(true); 
                return;
            }

            // Fall-off safety trigger
            if (player.y > canvas.height + 20) {
                triggerInstantDeath("Fell down the waterfall gap!");
            }

            // Invulnerability blink clock decrementer
            if (player.isInvulnerable) {
                player.invulnTimer -= 0.016;
                if (player.invulnTimer <= 0) {
                    player.isInvulnerable = false;
                }
            }

            // Test collisions against active historical stations
            if (!ninjaAttackActive) {
                for (let i = 0; i < quizStations.length; i++) {
                    let st = quizStations[i];
                    if (st.active && isColliding(player, st)) {
                        triggerQuiz(st);
                        break; 
                    }
                }
            }

            // Damage resolution from project threats
            if (!player.isInvulnerable && !player.isDying) {
                projectiles.forEach(p => {
                    let hit = false;
                    if (p.type === 'shuriken') {
                        let dist = Math.hypot(p.x - (player.x + player.width/2), p.y - (player.y + player.height/2));
                        if (dist < p.radius + 14) hit = true;
                    } else if (p.type === 'zero_plane') {
                        if (isColliding(player, { x: p.x, y: p.y, width: p.width || 55, height: p.height || 22 })) {
                            hit = true;
                        }
                    }

                    if (hit) {
                        triggerDirectInjury();
                    }
                });
            }
        }

        function triggerDirectInjury() {
            try {
                sfx.playHurt();
            } catch (e) {}
            lives--;
            updateHUD();

            if (lives <= 0) {
                triggerInstantDeath("Demolished by aerial warfare!");
            } else {
                player.isInvulnerable = true;
                player.invulnTimer = 2.0; 
                triggerFlashMessage("REDUCED BY TRAFFIC!", "bg-orange-600");
                player.vx = -8;
                player.vy = -4;
            }
        }

        // --- FLASH FEEDBACK OVERLAYS ---
        function triggerInstantDeath(reason) {
            if (player.isDying) return;
            player.isDying = true;
            player.vy = -12;

                sfx.playWrong();

        function handleDeathRespawn() {
            lives--;
            updateHUD();

            if (lives <= 0) {
                concludeGame(false); 
            } else {
                player.isDying = false;
                player.isInvulnerable = true;
                player.invulnTimer = 3.0;
                player.y = 150;
                player.vy = 0;
                player.vx = 0;

                let marioProgress = player.x / LEVEL_WIDTH;
                if (marioProgress < 0.25) {
                    player.x = 100;
                } else if (marioProgress < 0.50) {
                    player.x = 1350;
                } else if (marioProgress < 0.75) {
                    player.x = 2450;
                } else {
                    player.x = 3950;
                }

                ninjaAttackActive = false;
                ninja = null;
            }
        }

        function triggerQuiz(station) {

        function submitAnswer(selectedIndex) {
            if (!quizModal || quizModal.classList.contains('hidden')) return;
            
            quizModal.classList.add('hidden');
            isPaused = false;
            
            if (!currentActiveStation) return;
            let q = currentActiveStation.assignedQuestion;
            if (!q) {
                currentActiveStation.active = false;
                currentActiveStation = null;
                return;
            }
            
            if (selectedIndex === q.correct) {
                try {
                    sfx.playCorrect();
                } catch (e) {}
                coins += 10;
                questionsAnsweredCount++;
                updateHUD();
                
                // Spawn sparkles
                for (let i = 0; i < 5; i++) {
                    floatingScores.push({
                        x: player.x + Math.random() * 40 - 20,
                        y: player.y - 30 + Math.random() * 30,
                        text: "💮",
                        timer: 1.2
                    });
                }

                currentActiveStation.active = false;

                // Push facts to UI
                latestUnlockedArticle = q.article;
                unlockedArticlesList.push(q);
                updateArchivesModalList();

                showCorrectFeedback();
                currentActiveStation = null; // Clear active referencer to avoid overlapping calls
            } else {
                // ADAPTIVE RETRY SYSTEM (REPEAT UNTIL GETTING IT RIGHT)
                let futureActive = quizStations.filter(st => st.active && st.x > currentActiveStation.x);
                
                if (futureActive.length >= 2) {
                    let failedQ = currentActiveStation.assignedQuestion;
                    let tempQ = futureActive[1].assignedQuestion;
                    futureActive[1].assignedQuestion = failedQ;
                    
                    for (let j = 2; j < futureActive.length; j++) {
                        let nextTemp = futureActive[j].assignedQuestion;
                        futureActive[j].assignedQuestion = tempQ;
                        tempQ = nextTemp;
                    }
                } else if (futureActive.length === 1) {
                    futureActive[0].assignedQuestion = currentActiveStation.assignedQuestion;
                } else {
                    let spawnX = player.x + 180;
                    if (spawnX < LEVEL_WIDTH - 200) {
                        quizStations.push({
                            x: spawnX,
                            y: 305,
                            width: 32,
                            height: 32,
                            type: 'sushi',
                            emoji: '🍥',
                            assignedQuestion: currentActiveStation.assignedQuestion,
                            active: true,
                            bounce: Math.random() * Math.PI,
                            jumpTimer: Math.random(),
                            originalY: 305,
                            label: "Retry!"
                        });
                    }
                }

                currentActiveStation.active = false;
                currentActiveStation = null; // Clear active referencer to avoid overlapping calls
                spawnNinjaAttack();
            }
        }

        // --- CORE GAME MAIN LOOP ---
        function gameLoop(timestamp) {
            if (!isRunning) return;

            if (!lastTime) lastTime = timestamp;
            
            let elapsed = timestamp - lastTime;
            lastTime = timestamp;

            if (elapsed > 100) elapsed = 100;

            if (!isPaused) {
                accumulator += elapsed;
                
                while (accumulator >= PHYSICS_DELTATIME) {
                    updatePlayerPhysics();
                    updateProjectiles();
                    updateSakura();
                    if (ninjaAttackActive) {
                        updateNinja();
                    }
                    accumulator -= PHYSICS_DELTATIME;
                }

                ctx.clearRect(0, 0, canvas.width, canvas.height);

                let viewX = player.x - 250;
                if (viewX < 0) viewX = 0;
                if (viewX > LEVEL_WIDTH - canvas.width) viewX = LEVEL_WIDTH - canvas.width;
                
                const viewXInt = Math.floor(viewX);

                drawSkyBackground(player.x);
                drawSakura();
                drawPlatforms(viewXInt);
                drawQuizStations(viewXInt);
                drawProjectiles(viewXInt);
                
                if (ninjaAttackActive) {
                    drawNinja(viewXInt);
                }

                drawPlayer(viewXInt);
                drawFloatingScores(viewXInt);

                drawParchmentScrollOnMountain();
                drawCanvasTouchControls();
            } else {
                lastTime = timestamp;
            }

            requestAnimationFrame(gameLoop);
        }

        // --- MOBILE/PORTRAIT RESPONSIVE CANVAS-DIRECT TOUCH SYSTEM ---
        function handleCanvasTouch(e) {
            if (!isRunning || isPaused || player.isDying) return;
            e.preventDefault();

            const rect = canvas.getBoundingClientRect();
            keys.left = false;
            keys.right = false;

            for (let i = 0; i < e.touches.length; i++) {
                let touch = e.touches[i];
                let tx = ((touch.clientX - rect.left) / rect.width) * 800;
                let ty = ((touch.clientY - rect.top) / rect.height) * 400;

                // Translucent left paddle
                let distLeft = Math.hypot(tx - 60, ty - 345);
                if (distLeft < 45) {
                    keys.left = true;
                    continue;
                }

                // Translucent right paddle
                let distRight = Math.hypot(tx - 140, ty - 345);
                if (distRight < 45) {
                    keys.right = true;
                    continue;
                }

                // Translucent jump paddle
                let distJump = Math.hypot(tx - 730, ty - 345);
                if (distJump < 50) {
                    if (player.grounded && !player.isDying) {
                        player.vy = player.jumpPower;
                        player.grounded = false;
                        try {
                            sfx.playJump();
                        } catch (err) {}
                    }
                }
            }
        }

        function handleCanvasTouchEnd(e) {
            e.preventDefault();
            if (e.touches.length === 0) {
                keys.left = false;
                keys.right = false;
            } else {
                handleCanvasTouch(e);
            }
        }
