let shuffledQuestions = [];
let currentQuestion = null;
let asking = false;
let fireworks = [];

const player = { x:80, y:0, w:38, h:52, vx:0, vy:0, speed:5.2, jump:-14, grounded:false };

let groundY, blocks = [], cameraX = 0;
let score = 0, lives = 5, qAnswered = 0, timeLeft = 180;
let gameStarted = false, gameOver = false, timerInterval;

const keys = { left:false, right:false, jump:false };

function shuffle(arr) {
  return arr.map(value => ({value, sort:Math.random()}))
            .sort((a,b) => a.sort - b.sort)
            .map(item => item.value);
}

function makeWorld() {
  groundY = canvas.height * 0.7;
  blocks = [];
  for (let i = 0; i < 50; i++) {
    blocks.push({ x:250 + i * 210, y:groundY - 150 - (i % 3) * 22, w:48, h:48, used:false });
  }
  player.y = groundY - player.h;
}

function startGame() {
  document.getElementById("startScreen").style.display = "none";
  gameStarted = true;
  shuffledQuestions = shuffle([...questions]);
  makeWorld();

  timerInterval = setInterval(() => {
    if (!asking && !gameOver) {
      timeLeft--;
      document.getElementById("time").textContent = timeLeft;
      if (timeLeft <= 0) endGame();
    }
  }, 1000);
}

function drawBackground() {
  ctx.fillStyle = "#ffffff";
  ctx.font = "bold 30px Arial";
  ctx.fillText("🇸🇬 Happy National Day Singapore 🇸🇬", 35, 85);

  ctx.fillStyle = "#e60012";
  for (let i = 0; i < 9; i++) {
    let x = i * 260 - (cameraX * 0.3 % 260);
    ctx.fillRect(x, groundY - 58, 84, 58);
    ctx.fillStyle = "#fff";
    ctx.fillRect(x + 10, groundY - 43, 18, 18);
    ctx.fillRect(x + 48, groundY - 43, 18, 18);
    ctx.fillStyle = "#e60012";
  }

  ctx.fillStyle = "#36a852";
  ctx.fillRect(0, groundY, canvas.width, canvas.height - groundY);
}

function drawBlocks() {
  for (const b of blocks) {
    const bx = b.x - cameraX;
    if (bx < -80 || bx > canvas.width + 80) continue;

    ctx.fillStyle = b.used ? "#999" : "#ffcc00";
    ctx.fillRect(bx, b.y, b.w, b.h);
    ctx.strokeStyle = "#111";
    ctx.lineWidth = 3;
    ctx.strokeRect(bx, b.y, b.w, b.h);

    ctx.fillStyle = "#111";
    ctx.font = "bold 29px Arial";
    ctx.fillText(b.used ? "✓" : "?", bx + 14, b.y + 34);
  }
}

function drawPlayer() {
  const px = player.x - cameraX;

  ctx.fillStyle = "#ffcc99";
  ctx.fillRect(px + 8, player.y - 12, 24, 18);
  ctx.fillStyle = "#e60012";
  ctx.fillRect(px + 3, player.y - 22, 32, 11);
  ctx.fillRect(px, player.y, player.w, player.h);
  ctx.fillStyle = "#ffffff";
  ctx.fillRect(px + 7, player.y + 8, 24, 12);
  ctx.fillStyle = "#111";
  ctx.fillRect(px + 7, player.y + 25, 9, 24);
  ctx.fillRect(px + 22, player.y + 25, 9, 24);
}

function update() {
  if (!gameStarted || gameOver || asking) return;

  player.vx = 0;
  if (keys.left) player.vx = -player.speed;
  if (keys.right) player.vx = player.speed;

  if (keys.jump && player.grounded) {
    player.vy = player.jump;
    player.grounded = false;
  }

  player.vy += 0.7;
  player.x += player.vx;
  player.y += player.vy;

  if (player.y + player.h >= groundY) {
    player.y = groundY - player.h;
    player.vy = 0;
    player.grounded = true;
  }

  if (player.x < 0) player.x = 0;
  cameraX = Math.max(0, player.x - canvas.width * 0.35);

  for (const b of blocks) {
    if (b.used) continue;

    const hit =
      player.x < b.x + b.w &&
      player.x + player.w > b.x &&
      player.y < b.y + b.h &&
      player.y + player.h > b.y;

    if (hit && player.vy < 0) {
      b.used = true;
      player.vy = 5;
      askQuestion();
      break;
    }
  }
}

function askQuestion() {
  if (qAnswered >= 50) {
    endGame();
    return;
  }

  asking = true;
  currentQuestion = shuffledQuestions[qAnswered];

  document.getElementById("qText").textContent = currentQuestion.q;
  document.getElementById("feedback").style.display = "none";
  document.getElementById("feedback").textContent = "";

  showOptions();
  document.getElementById("questionBox").style.display = "flex";
}

function showOptions() {
  const optionsDiv = document.getElementById("options");
  optionsDiv.innerHTML = "";

  const allOptions = shuffle([
    { text: currentQuestion.a, correct: true },
    ...currentQuestion.o.map(x => ({ text:x, correct:false }))
  ]);

  allOptions.forEach(opt => {
    const btn = document.createElement("button");
    btn.className = "option";
    btn.textContent = opt.text;
    btn.onclick = () => answer(opt.correct);
    optionsDiv.appendChild(btn);
  });
}

function answer(correct) {
  if (correct) {
    score += 10;
    qAnswered++;

    document.getElementById("score").textContent = score;
    document.getElementById("qCount").textContent = qAnswered;
    document.getElementById("questionBox").style.display = "none";
    asking = false;

    if (qAnswered >= 50) endGame();
  } else {
    lives--;

    document.getElementById("lives").textContent = lives;
    document.getElementById("feedback").style.display = "block";
    document.getElementById("feedback").innerHTML =
      "❌ Not quite. Correct answer: <br><span style='font-size:22px;color:#e60012;'>" +
      currentQuestion.a +
      "</span><br>Try the same question again.";

    showOptions();

    if (lives <= 0) {
      setTimeout(endGame, 1200);
    }
  }
}

function createFirework() {
  const x = Math.random() * canvas.width;
  const y = Math.random() * canvas.height * 0.45;
  for (let i = 0; i < 35; i++) {
    fireworks.push({
      x, y,
      vx: Math.cos(i) * Math.random() * 5,
      vy: Math.sin(i) * Math.random() * 5,
      life: 60,
      color: Math.random() > 0.5 ? "#ffffff" : "#ffcc00"
    });
  }
}

function drawFireworks() {
  if (!gameOver) return;

  if (Math.random() < 0.08) createFirework();

  fireworks.forEach(p => {
    ctx.fillStyle = p.color;
    ctx.beginPath();
    ctx.arc(p.x, p.y, 3, 0, Math.PI * 2);
    ctx.fill();

    p.x += p.vx;
    p.y += p.vy;
    p.vy += 0.03;
    p.life--;
  });

  fireworks = fireworks.filter(p => p.life > 0);
}

function endGame() {
  gameOver = true;
  clearInterval(timerInterval);

  document.getElementById("questionBox").style.display = "none";
  document.getElementById("endScreen").style.display = "flex";
  document.getElementById("finalScore").textContent = `Score: ${score} / 500`;

  if (score >= 420) {
    document.getElementById("finalMsg").textContent = "Solid lah. You are very Singapore-ready.";
  } else if (score >= 280) {
    document.getElementById("finalMsg").textContent = "Not bad. Replay once and become more power.";
  } else {
    document.getElementById("finalMsg").textContent = "Can improve. Try again and chope a better score.";
  }
}

function draw() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  drawBackground();
  drawBlocks();
  drawPlayer();
  drawFireworks();
}

function loop() {
  update();
  draw();
  requestAnimationFrame(loop);
}
loop();

window.addEventListener("keydown", e => {
  if (e.key === "ArrowLeft" || e.key.toLowerCase() === "a") keys.left = true;
  if (e.key === "ArrowRight" || e.key.toLowerCase() === "d") keys.right = true;
  if (e.key === " " || e.key === "ArrowUp" || e.key.toLowerCase() === "w") keys.jump = true;
});

window.addEventListener("keyup", e => {
  if (e.key === "ArrowLeft" || e.key.toLowerCase() === "a") keys.left = false;
  if (e.key === "ArrowRight" || e.key.toLowerCase() === "d") keys.right = false;
  if (e.key === " " || e.key === "ArrowUp" || e.key.toLowerCase() === "w") keys.jump = false;
});

function hold(button, key) {
  button.addEventListener("touchstart", e => {
    e.preventDefault();
    keys[key] = true;
  });
  button.addEventListener("touchend", e => {
    e.preventDefault();
    keys[key] = false;
  });
  button.addEventListener("mousedown", () => keys[key] = true);
  button.addEventListener("mouseup", () => keys[key] = false);
  button.addEventListener("mouseleave", () => keys[key] = false);
}

hold(document.getElementById("left"), "left");
hold(document.getElementById("right"), "right");
hold(document.getElementById("jump"), "jump");
