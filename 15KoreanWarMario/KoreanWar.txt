function shuffle(a) { return a.sort(() => Math.random() - 0.5); }

function initStars() {
    state.stars = [];
    for(let i=0; i<40; i++) {
        state.stars.push({
            x: Math.random() * state.viewportW,
            y: Math.random() * state.viewportH,

function resize() {
    state.viewportW = window.innerWidth;
    state.viewportH = window.innerHeight;
    canvas.width = state.viewportW;
    canvas.height = state.viewportH;
    player.x = state.viewportW / 2 - player.w / 2;
    initStars();

const touchInput = { left: false, right: false };

async function initTracking() {
    if (!firebaseEnabled) return;
    // FIXED: Segment segment count must be even. path: artifacts/id/public/data/usage/stats
    const analyticsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    
    try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
            await signInWithCustomToken(auth, __initial_auth_token);
        } else {
            await signInAnonymously(auth);
        }
    } catch (err) {
        console.error("Auth failed", err);
    }
    
    onAuthStateChanged(auth, (user) => {
        state.user = user;
        if (user) {
            onSnapshot(analyticsDoc, (snap) => {
                if (snap.exists()) {
                    const count = snap.data().playCount || 0;
                    document.getElementById('global-play-count').innerText = count.toLocaleString();
                    document.getElementById('play-tracker').style.opacity = '1';
                } else {
                    setDoc(analyticsDoc, { playCount: 0 });
                }
            }, (err) => console.log("Tracking restricted"));
        }
    });
}

async function recordDeployment() {
    if (!firebaseEnabled || !state.user) return;
    const analyticsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    try {
        await updateDoc(analyticsDoc, { playCount: increment(1) });
    } catch (e) {
        await setDoc(analyticsDoc, { playCount: 1 }, { merge: true });
    }
}

function init() {
    resize();
    initTracking();
    
    currentSessionDeck = shuffle([...questionsDeckRaw]);

    document.getElementById('btn-start').onclick = () => {
        if (firebaseEnabled) recordDeployment();
        document.getElementById('start-screen').classList.add('hidden');
        state.gameStarted = true; 
        state.gameActive = true;
        state.lastTimeUpdate = performance.now();
    };
    document.getElementById('btn-restart').onclick = () => location.reload();

    const setupTouch = (id, key) => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { 
            e.preventDefault(); 
            touchInput[key] = true; 
        }, { passive: false });
        el.addEventListener('touchend', (e) => { 
            e.preventDefault(); 
            touchInput[key] = false; 
        }, { passive: false });
    };
    setupTouch('ctrl-left', 'left');
    setupTouch('ctrl-right', 'right');
    
    document.getElementById('ctrl-shoot').addEventListener('touchstart', (e) => {
        e.preventDefault();
        const now = performance.now();
        if (state.gameActive && !state.paused && now - state.lastShootTime > state.shootCooldown) {
            shoot();
            state.lastShootTime = now;
        }
    }, { passive: false });

    requestAnimationFrame(gameLoop);
}

function shoot() {
    state.bullets.push({ x: player.x + player.w/2 - 4, y: state.viewportH - 180, r: 10, speed: 15 });
}

function spawnEnemy() {
    const r = 38;
    state.enemies.push({
        x: Math.random() * (state.viewportW - r*2) + r,
        y: -r,
        r: r,
        speed: 2.2 + Math.random() * 1.5,
        type: 'mcq'
    });
}

function createFireworks(x, y) {
    const colors = ['#22c55e', '#fbbf24', '#3b82f6', '#ef4444', '#ffffff'];
    for (let i = 0; i < 35; i++) {
        state.particles.push({
            x: x, y: y,
            vx: (Math.random() - 0.5) * 15,
            vy: (Math.random() - 0.5) * 15,
            life: 1.0,
            color: colors[Math.floor(Math.random() * colors.length)],
            size: Math.random() * 3 + 1
        });
    }
}

function triggerQuiz() {
    if (state.paused) return; 
    state.paused = true;
    state.hitOccurred = false;
    
    const q = currentSessionDeck[0];
    const modal = document.getElementById('quiz-overlay');
    const container = document.getElementById('options-container');
    const feedback = document.getElementById('feedback');
    const qCount = document.getElementById('stats-q-remaining');
    
    qCount.innerText = `UNITS REMAINING: ${currentSessionDeck.length}`;
    document.getElementById('question-text').innerText = q.q;
    container.innerHTML = '';
    feedback.innerText = '';
    
    shuffle([...q.o]).forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn';
        btn.innerText = opt;
        btn.onclick = () => {
            const btns = Array.from(container.children);
            btns.forEach(b => b.disabled = true);
            
            if (opt === q.a) {
                btn.classList.add('correct-flash');
                state.score += 100;
                createFireworks(state.viewportW / 2, state.viewportH / 2);
                currentSessionDeck.shift();
                updateHUD();
                
                setTimeout(() => {
                    if (currentSessionDeck.length === 0) endGame(true);
                    else resume();
                }, 800);
            } else {
                btn.classList.add('wrong-flash');
                btns.forEach(b => {
                    if(b.innerText === q.a) b.classList.add('correct-flash');
                });
                
                feedback.innerText = `REVISION: The correct entry is ${q.a}`;
                feedback.className = "mt-8 text-center font-black uppercase italic text-red-500 text-sm leading-tight";
                
                const missed = currentSessionDeck.shift();
                const targetPos = Math.min(3, currentSessionDeck.length);
                currentSessionDeck.splice(targetPos, 0, missed);
                
                setTimeout(() => {
                    state.lives--;
                    updateHUD();
                    if (state.lives <= 0) endGame(false);
                    else resume();
                }, 2800);
            }
        };
        container.appendChild(btn);
    });
    modal.classList.remove('hidden');
}

function resume() {
    document.getElementById('quiz-overlay').classList.add('hidden');
    state.paused = false;
}

function gameLoop(now) {
    if (!state.gameStarted) { requestAnimationFrame(gameLoop); return; }

    if (!state.paused && state.gameActive) {
        if (now - state.lastTimeUpdate >= 1000) {
            state.timer--;
            state.lastTimeUpdate = now;
            document.getElementById('stats-timer').innerText = state.timer + "s";
            if (state.timer <= 0) endGame(false);
        }

        if (touchInput.left && player.x > 0) player.x -= player.speed;
        if (touchInput.right && player.x < state.viewportW - player.w) player.x += player.speed;

        if (now - state.lastSpawn > 2500) {
            spawnEnemy();
            state.lastSpawn = now;
        }

        // Clean up and update bullets
        for (let i = state.bullets.length - 1; i >= 0; i--) {
            const b = state.bullets[i];
            b.y -= b.speed;
            if (b.y < -50) state.bullets.splice(i, 1);
        }

        // Clean up and update enemies
        for (let i = state.enemies.length - 1; i >= 0; i--) {
            const en = state.enemies[i];
            en.y += en.speed;
            
            if (en.y > state.viewportH + en.r) {
                state.enemies.splice(i, 1);
                state.lives--;
                updateHUD();
                if (state.lives <= 0) endGame(false);
                continue;
            }

            // High performance collision
            for (let j = state.bullets.length - 1; j >= 0; j--) {
                const b = state.bullets[j];
                const dx = b.x - en.x;
                const dy = b.y - en.y;
                const distSq = dx * dx + dy * dy;
                // bullet hit detection
                if (distSq < (en.r + 15) * (en.r + 15)) {
                    state.bullets.splice(j, 1);
                    state.enemies.splice(i, 1);
                    state.hitOccurred = true;
                    break;
                }
            }
            if (state.hitOccurred) break;
        }

        if (state.hitOccurred) {
            triggerQuiz();
        }
    }

    // Stars Update
    state.stars.forEach(s => {
        s.y += s.speed;
        if (s.y > state.viewportH) {
            s.y = -20;
            s.x = Math.random() * state.viewportW;
        }
    });

    // Particles Update
    for (let i = state.particles.length - 1; i >= 0; i--) {
        const p = state.particles[i];
        p.x += p.vx; p.y += p.vy;
        p.life -= 0.02;
        if (p.life <= 0) state.particles.splice(i, 1);
    }

    render();
    requestAnimationFrame(gameLoop);
}

function render() {
    ctx.clearRect(0, 0, state.viewportW, state.viewportH);
    
    ctx.fillStyle = "rgba(255,255,255,0.4)";
    state.stars.forEach(s => {
        ctx.fillRect(s.x, s.y, s.size, s.size);
    });

    // Unit
    ctx.fillStyle = "#16a34a";
    ctx.fillRect(player.x, state.viewportH - 160, player.w, 45);
    ctx.fillStyle = "#15803d";
    ctx.fillRect(player.x + (player.w * 0.25), state.viewportH - 185, player.w * 0.5, 25);

    // Targets
    ctx.font = "900 24px Inter";
    ctx.textAlign = "center";
    state.enemies.forEach(en => {
        ctx.beginPath();
        ctx.arc(en.x, en.y, en.r, 0, Math.PI * 2);
        ctx.fillStyle = "#1e293b";
        ctx.fill();
        ctx.strokeStyle = "#22c55e";
        ctx.lineWidth = 4;
        ctx.stroke();
        ctx.fillStyle = "#ffffff";
        ctx.fillText("?", en.x, en.y + 8);
    });

    // Bullets
    ctx.fillStyle = "#fbbf24";
    state.bullets.forEach(b => {
        ctx.fillRect(b.x, b.y, 8, 22);
    });

    // FX
    state.particles.forEach(p => {
        ctx.globalAlpha = p.life;
        ctx.fillStyle = p.color;
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fill();
    });
    ctx.globalAlpha = 1.0;
}

function endGame(victory) {
    state.gameActive = false;
    const title = document.getElementById('end-title');
    title.innerText = victory ? "FACTS MASTERED" : "UNIT DEFEATED";
    title.className = victory ? "text-6xl loud-font italic text-green-600 mb-8 uppercase" : "text-6xl loud-font italic text-red-600 mb-8 uppercase";
    document.getElementById('final-score').innerText = state.score;
    document.getElementById('end-screen').classList.remove('hidden');
}
