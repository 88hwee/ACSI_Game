function shuffle(a) {
    let array = [...a];
    for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
    }
    return array;
}

async function initTracking() {
    if (!firebaseEnabled) return;
    const analyticsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');

    const initAuth = async () => {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
            await signInWithCustomToken(auth, __initial_auth_token);
        } else {
            await signInAnonymously(auth);
        }
    };
    
    await initAuth();

    onAuthStateChanged(auth, async (user) => {
        if (user) {
            state.user = user;
            onSnapshot(analyticsDoc, (snapshot) => {
                if (snapshot.exists()) {
                    const count = snapshot.data().playCount || 0;
                    document.getElementById('global-play-count').innerText = count.toLocaleString();
                    document.getElementById('play-tracker').style.opacity = '1';
                } else {
                    setDoc(analyticsDoc, { playCount: 0 });
                }
            }, (err) => console.log("Tracking restricted"));
        }
    });
}

async function recordDeployment() {
    if (!firebaseEnabled || !state.user) return;
    const analyticsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    try {
        await updateDoc(analyticsDoc, { playCount: increment(1) });
    } catch (e) {
        await setDoc(analyticsDoc, { playCount: 1 }, { merge: true });
    }
}

function initBlocks() {
    blocks = [];
    for (let i = 0; i < 70; i++) {
        blocks.push({
            x: 800 + (i * 600),
            y: state.groundY - 240 - (Math.random() * 120),
            w: 80, h: 80,
            active: true,
            type: 'quiz'
        });
        if (Math.random() > 0.4) {
            blocks.push({
                x: 1100 + (i * 600),
                y: state.groundY - 140,
                w: 80, h: 80,
                active: true,
                type: 'brick'
            });
        }
    }
}

function resize() {
    state.viewportW = window.innerWidth;
    state.viewportH = window.innerHeight;
    canvas.width = state.viewportW;
    canvas.height = state.viewportH;
    state.groundY = state.viewportH - 100;
    
    if (!state.gameStarted) {
        player.y = state.groundY - player.h;
        initBlocks();
    }
}

function handleJump() {
    if (!state.gameActive || state.paused) return;
    if (player.jumpsAvailable > 0) {
        player.vy = player.jumpPower;
        player.jumpsAvailable--;
        player.grounded = false;
    }
}

function init() {
    window.addEventListener('resize', resize);
    window.addEventListener('orientationchange', () => setTimeout(resize, 200));
    resize();
    initTracking();

    sessionDeck = shuffle([...KOREAN_WAR_QUESTIONS]);

    // High-precision keyboard listeners using e.code for laptop compatibility
    const handleKey = (e, isDown) => {
        if (!state.gameActive || state.paused) return;
        const code = e.code;
        
        // Movement
        if (code === 'ArrowLeft' || code === 'KeyA') { e.preventDefault(); inputs.left = isDown; }
        if (code === 'ArrowRight' || code === 'KeyD') { e.preventDefault(); inputs.right = isDown; }
        
        // Jump
        if (isDown && (code === 'Space' || code === 'KeyW' || code === 'ArrowUp')) {
            e.preventDefault();
            handleJump();
        }
    };
    window.addEventListener('keydown', (e) => handleKey(e, true));
    window.addEventListener('keyup', (e) => handleKey(e, false));

    // Force focus on canvas click for laptop
    canvas.addEventListener('mousedown', () => {
        canvas.focus();
    });

    const setupTouch = (id, key) => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { 
            e.preventDefault(); 
            inputs[key] = true; 
        }, { passive: false });
        el.addEventListener('touchend', (e) => { 
            e.preventDefault(); 
            inputs[key] = false; 
        }, { passive: false });
    };
    setupTouch('ctrl-left', 'left');
    setupTouch('ctrl-right', 'right');
    document.getElementById('ctrl-jump').addEventListener('touchstart', (e) => { 
        e.preventDefault(); 
        handleJump(); 
    }, { passive: false });

    document.getElementById('btn-start').onclick = () => {
        if (firebaseEnabled) recordDeployment();
        document.getElementById('start-screen').classList.add('hidden');
        state.gameStarted = true;
        state.gameActive = true;
        
        // Transfer focus to game immediately
        canvas.focus();
        window.focus();
    };

    document.getElementById('btn-restart').onclick = () => location.reload();
    requestAnimationFrame(update);
}

function triggerQuiz() {
    if (state.paused) return;
    state.paused = true;
    const q = sessionDeck[0];
    const modal = document.getElementById('quiz-overlay');
    const container = document.getElementById('options-container');
    const feedback = document.getElementById('feedback-area');
    
    document.getElementById('question-text').innerText = q.q;
    document.getElementById('stats-q-remaining').innerText = `DATA REQUISITIONS: ${sessionDeck.length}`;
    feedback.classList.add('hidden');
    container.classList.remove('hidden');
    container.innerHTML = '';
    
    const choices = shuffle([...q.o]);
    
    choices.forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn'; 
        btn.innerText = opt;
        btn.onclick = () => {
            const btns = Array.from(container.children);
            btns.forEach(b => b.disabled = true);
            
            if (opt === q.a) {
                btn.classList.add('correct-flash');
                state.score += 1000;
                document.getElementById('stats-score').innerText = state.score;
                sessionDeck.shift();
                
                feedback.classList.remove('hidden');
                container.classList.add('hidden');
                feedback.className = "mt-6 p-5 rounded-[25px] border-4 border-green-500 bg-green-50";
                document.getElementById('feedback-title').innerText = "VERIFIED";
                document.getElementById('feedback-title').className = "font-black uppercase text-xl mb-1 text-green-600";
                document.getElementById('feedback-detail').innerText = "Data intelligence verified. Ready for next phase.";

                if (sessionDeck.length === 0) {
                    setTimeout(() => endGame(true), 1200);
                } else {
                    setTimeout(() => resume(), 1200);
                }
            } else {
                btn.classList.add('wrong-flash');
                btns.forEach(b => { if(b.innerText === q.a) b.classList.add('correct-flash'); });
                
                state.lives--;
                updateLivesHUD();
                
                feedback.classList.remove('hidden');
                container.classList.add('hidden');
                feedback.className = "mt-6 p-5 rounded-[25px] border-4 border-red-500 bg-red-50";
                document.getElementById('feedback-title').innerText = "CORRUPTED";
                document.getElementById('feedback-title').className = "font-black uppercase text-xl mb-1 text-red-600";
                document.getElementById('feedback-detail').innerText = `Fact: ${q.a}. Re-queueing mission.`;

                const missedQ = sessionDeck.shift();
                sessionDeck.splice(Math.min(3, sessionDeck.length), 0, missedQ); 

                setTimeout(() => {
                    if (state.lives <= 0) endGame(false);
                    else resume();
                }, 2800);
            }
        };
        container.appendChild(btn);
    });
    modal.classList.remove('hidden');
}

function updateLivesHUD() {
    document.getElementById('stats-lives').innerText = '❤️'.repeat(Math.max(0, state.lives)) + '🖤'.repeat(Math.max(0, 5 - state.lives));
}

function resume() {
    document.getElementById('quiz-overlay').classList.add('hidden');
    state.paused = false;
    canvas.focus(); // Return focus to game after quiz
}

function update() {
    if (state.gameStarted && state.gameActive && !state.paused) {
        if (inputs.right) { player.vx = player.speed; player.facingRight = true; player.frame += 0.25; }
        else if (inputs.left) { player.vx = -player.speed; player.facingRight = false; player.frame += 0.25; }
        else { player.vx *= 0.85; player.frame = 0; }

        player.vy += 1.0; 
        player.x += player.vx;
        player.y += player.vy;

        if (player.y + player.h > state.groundY) { 
            player.y = state.groundY - player.h;
            player.vy = 0; 
            player.grounded = true;
            player.jumpsAvailable = 2;
        }

        blocks.forEach(b => {
            if (!b.active && b.type === 'quiz') return;
            if (player.x < b.x + b.w && player.x + player.w > b.x && 
                player.y < b.y + b.h && player.y + player.h > b.y) {
                if (player.vy < 0 && player.y > b.y + b.h - 35) {
                    player.vy = 5;
                    if (b.type === 'quiz' && b.active) {
                        b.active = false;
                        triggerQuiz();
                    }
                } 
                else if (player.vy > 0 && player.y + player.h < b.y + 35) {
                    player.y = b.y - player.h;
                    player.vy = 0;
                    player.grounded = true;
                    player.jumpsAvailable = 2;
                }
                else if (player.vx > 0 && player.x + player.w < b.x + 25) { player.x = b.x - player.w; }
                else if (player.vx < 0 && player.x > b.x + b.w - 25) { player.x = b.x + b.w; }
            }
        });

        if (player.x < 0) player.x = 0;
        let targetCamX = -player.x + (state.viewportW * 0.25);
        state.cameraX += (targetCamX - state.cameraX) * 0.15;
        if (state.cameraX > 0) state.cameraX = 0;
    }
    render();
    requestAnimationFrame(update);
}

function render() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    ctx.fillStyle = "#5c94fc";
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    
    // Background
    ctx.fillStyle = "rgba(255,255,255,0.4)";
    for(let i=0; i<15; i++) {
        ctx.beginPath(); ctx.ellipse(400 + i*900, 180, 100, 50, 0, 0, Math.PI*2); ctx.fill();
    }
    
    ctx.save();
    ctx.translate(state.cameraX, 0);
    
    // Level
    ctx.fillStyle = "#16a34a"; ctx.fillRect(-5000, state.groundY, 80000, 30);
    ctx.fillStyle = "#5c4033"; ctx.fillRect(-5000, state.groundY + 30, 80000, 150);
    
    blocks.forEach(b => { 
        if (!b.active && b.type === 'quiz') {
            ctx.fillStyle = '#6b7280'; ctx.beginPath(); 
            if (ctx.roundRect) ctx.roundRect(b.x, b.y, b.w, b.h, 12);
            else ctx.rect(b.x, b.y, b.w, b.h);
            ctx.fill();
            return;
        }
        if (b.type === 'quiz') {
            ctx.fillStyle = '#fbbf24'; ctx.beginPath(); 
            if (ctx.roundRect) ctx.roundRect(b.x, b.y, b.w, b.h, 12);
            else ctx.rect(b.x, b.y, b.w, b.h);
            ctx.fill();
            ctx.fillStyle = '#0f172a'; ctx.font = 'black 50px Inter'; ctx.textAlign = 'center';
            ctx.fillText('?', b.x + b.w/2, b.y + b.h/2 + 18);
        } else {
            ctx.fillStyle = '#15803d'; ctx.beginPath(); 
            if (ctx.roundRect) ctx.roundRect(b.x, b.y, b.w, b.h, 6);
            else ctx.rect(b.x, b.y, b.w, b.h);
            ctx.fill();
        }
    });
    
    // Character
    ctx.save();
    ctx.translate(player.x + player.w / 2, player.y + player.h / 2);
    if (!player.facingRight) ctx.scale(-1, 1);
    const bounce = Math.sin(player.frame * 2) * 6;
    ctx.fillStyle = "#4b2c20"; ctx.fillRect(-18, -40 + bounce, 36, 20);
    ctx.fillStyle = "#ffdbac"; ctx.fillRect(-18, -20 + bounce, 36, 30);
    ctx.fillStyle = "#000"; ctx.fillRect(8, -12 + bounce, 5, 5);
    ctx.fillStyle = "#16a34a"; ctx.fillRect(-20, 10 + bounce, 40, 30);
    ctx.fillStyle = "#1e3a8a"; ctx.fillRect(-20, 25 + bounce, 40, 15);
    ctx.fillRect(-20, 40, 16, 18); ctx.fillRect(4, 40, 16, 18);
    ctx.restore();
    ctx.restore();
}

    state.gameActive = false;

    document.getElementById('final-score').innerText = state.score;
    document.getElementById('end-screen').classList.remove('hidden');
}
