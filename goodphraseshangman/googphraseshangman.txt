    // Canvas Fireworks Setup (Adapted for soft pastel theme background)
    const canvas = document.getElementById("fireworks-canvas");
    const ctx = canvas.getContext("2d");
    let particles = [];
    let isCelebrating = false;
    let animationId = null;

    function resizeCanvas() {
      canvas.width = window.innerWidth;
      canvas.height = window.innerHeight;
    }
    window.addEventListener('resize', resizeCanvas);
    resizeCanvas();

    // Particle class for beautiful fireworks blasts
    class Particle {
      constructor(x, y, color) {
        this.x = x;
        this.y = y;
        this.color = color;
        this.angle = Math.random() * Math.PI * 2;
        this.speed = Math.random() * 5 + 4;
        this.friction = 0.95;
        this.gravity = 0.12;
        this.alpha = 1;
        this.decay = Math.random() * 0.015 + 0.012;
        this.size = Math.random() * 3 + 1.5;
      }

      update() {
        this.speed *= this.friction;
        this.x += Math.cos(this.angle) * this.speed;
        this.y += Math.sin(this.angle) * this.speed + this.gravity;
        this.alpha -= this.decay;
      }

      draw() {
        ctx.save();
        ctx.globalAlpha = this.alpha;
        ctx.fillStyle = this.color;
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
        ctx.fill();
        ctx.restore();
      }
    }

    class Firework {
      constructor() {
        this.x = Math.random() * canvas.width;
        this.y = canvas.height;
        this.targetY = Math.random() * (canvas.height * 0.5) + 60;
        this.speed = Math.random() * 4 + 7;
        this.color = `hsl(${Math.random() * 360}, 100%, 65%)`;
        this.exploded = false;
      }

      update() {
        this.y -= this.speed;
        if (this.y <= this.targetY) {
          this.exploded = true;
          this.explode();
        }
      }

      draw() {
        if (!this.exploded) {
          ctx.beginPath();
          ctx.arc(this.x, this.y, 3, 0, Math.PI * 2);
          ctx.fillStyle = this.color;
          ctx.fill();
        }
      }

      explode() {
        const count = Math.floor(Math.random() * 30) + 50;
        for (let i = 0; i < count; i++) {
          particles.push(new Particle(this.x, this.y, this.color));
        }
      }
    }

    let activeFireworks = [];

    function startCelebration() {
      isCelebrating = true;
      activeFireworks = [];
      particles = [];

      // 1. Firework launcher intervals
      for (let i = 0; i < 5; i++) {
        setTimeout(() => {
          if (isCelebrating) {
            const x = Math.random() * (canvas.width - 200) + 100;
            const y = Math.random() * (canvas.height * 0.4) + 100;
            const colors = ['#e91e63', '#9c27b0', '#ffeb3b', '#00e676', '#00b0ff', '#ffd700', '#ff5722'];
            const randomColor = colors[Math.floor(Math.random() * colors.length)];
            
            const count = 70;
            for (let c = 0; c < count; c++) {
              particles.push(new Particle(x, y, randomColor));
            }
          }
        }, i * 300);
      }
      
      animate();
    }

    function stopCelebration() {
      isCelebrating = false;
      if (animationId) {
        cancelAnimationFrame(animationId);
      }
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      particles = [];
      activeFireworks = [];
    }

    function animate() {
      if (!isCelebrating && particles.length === 0 && activeFireworks.length === 0) {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        return;
      }

      // Transparent pastel-pink trail fits light background
      ctx.fillStyle = 'rgba(255, 238, 248, 0.25)'; 
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      if (isCelebrating && Math.random() < 0.06 && activeFireworks.length < 4) {
        activeFireworks.push(new Firework());
      }

      activeFireworks = activeFireworks.filter(fw => !fw.exploded);
      activeFireworks.forEach(fw => {
        fw.update();
        fw.draw();
      });

      particles = particles.filter(p => p.alpha > 0);
      particles.forEach(p => {
        p.update();
        p.draw();
      });

      animationId = requestAnimationFrame(animate);
    }

    // Hangman Gameplay Functions
    function nextWord() {
      stopCelebration();
      document.getElementById("celebration-modal-overlay").classList.remove("active");

      // Refill & Shuffle Pool if exhausted
      if (pool.length === 0) {
        pool = [...Array(words.length).keys()];
        pool.sort(() => Math.random() - 0.5);
      }

      const selectedIndex = pool.pop();
      const item = words[selectedIndex];
      
      chosenWord = item.word.toUpperCase();
      chosenClueEN = item.clueEN;
      chosenClueJA = item.clueJA;
      guessed = [];
      wrong = 0;

      // Render Bilingual clue layout
      document.getElementById("clue").innerHTML = `
        <div style="margin-bottom: 6px;">🌱 <strong>Clue Hint:</strong> ${chosenClueEN}</div>
        <div style="font-size: clamp(0.95rem, 2.5vw, 1.15rem); color: #7c6485; font-weight: 500; border-top: 1px dashed #ffd1dc; padding-top: 6px;">💡 <strong>ヒント:</strong> ${chosenClueJA}</div>
      `;
      
      const statusElement = document.getElementById("status");
      statusElement.textContent = "";
      statusElement.className = "";

      // Instantly hide all hangman stick figure parts
      svgParts.forEach(part => {
        part.classList.remove('visible');
      });

      createKeypad();
      updateDisplay();
    }

    // Creates the big visual keypad
    function createKeypad() {
      const lettersDiv = document.getElementById("letters");
      lettersDiv.innerHTML = "";

      "ABCDEFGHIJKLMNOPQRSTUVWXYZ".split("").forEach(letter => {
        const btn = document.createElement("button");
        btn.textContent = letter;
        btn.onclick = () => guessLetter(letter, btn);
        lettersDiv.appendChild(btn);
      });
    }

    function guessLetter(letter, btn) {
      if (guessed.includes(letter) || wrong >= svgParts.length) return;

      btn.disabled = true;
      guessed.push(letter);

      if (chosenWord.includes(letter)) {
        btn.classList.add("hit"); // Big green key
      } else {
        btn.classList.add("miss"); // Big red key

        // Render next step of stick figure
        if (svgParts[wrong]) {
          svgParts[wrong].classList.add('visible');
        }
        wrong++;
      }

      updateDisplay();
      checkGameStatus();
    }

    // Render exactly one word per line
    function updateDisplay() {
      const displayContainer = document.getElementById("word");
      displayContainer.innerHTML = "";

      // Split the overall phrase by spaces to separate individual words
      const wordsArray = chosenWord.split(" ");
      
      wordsArray.forEach((wordStr) => {
        const wordRow = document.createElement("div");
        wordRow.className = "word-row";
        
        wordStr.split("").forEach(char => {
          const letterSpan = document.createElement("span");
          
          if (char === "-" || char === "'" || char === "." || char === "!") {
            letterSpan.className = "word-letter punctuation";
            letterSpan.textContent = char;
          } else {
            letterSpan.className = "word-letter";
            letterSpan.textContent = guessed.includes(char) ? char : "_";
          }
          wordRow.appendChild(letterSpan);
        });
        
        displayContainer.appendChild(wordRow);
      });

      document.getElementById("score").textContent = score;
      document.getElementById("progress").textContent = `${unlockedWords.size}/${words.length}`;
    }

    function checkGameStatus() {
      const isWon = chosenWord
        .split("")
        .every(char => char === " " || char === "-" || char === "'" || char === "." || char === "!" || guessed.includes(char));

      const statusElement = document.getElementById("status");

      if (isWon) {
        score += 10;
        if (!unlockedWords.has(chosenWord)) {
          unlockedWords.add(chosenWord);
        }
        
        statusElement.textContent = "⭐️ 素晴らしい！ Excellent! ⭐️";
        statusElement.className = "won";
        disableAllButtons();
        
        // Ignite full screen fireworks!
        startCelebration();

        // Launch modal after a small lag
        setTimeout(() => {
          showCelebrationModal();
        }, 900);

      } else if (wrong >= svgParts.length) {
        statusElement.textContent = "Let's keep growing! Try again!";
        statusElement.className = "lost";
        
        // Show correct answer in highlighted red
        const displayContainer = document.getElementById("word");
        displayContainer.innerHTML = "";
        
        const wordsArray = chosenWord.split(" ");
        wordsArray.forEach((wordStr) => {
          const wordRow = document.createElement("div");
          wordRow.className = "word-row";
          
          wordStr.split("").forEach(char => {
            const letterSpan = document.createElement("span");
            letterSpan.className = "word-letter";
            letterSpan.style.color = "var(--error)";
            letterSpan.style.borderBottom = "none";
            letterSpan.textContent = char;
            wordRow.appendChild(letterSpan);
          });
          displayContainer.appendChild(wordRow);
        });

        disableAllButtons();
      }
    }

    function disableAllButtons() {
      document.querySelectorAll(".letters button").forEach(btn => btn.disabled = true);
    }

    // Modal Control functions
    function showCelebrationModal() {
      document.getElementById("modal-word").textContent = chosenWord;
      document.getElementById("modal-definition").innerHTML = `
        <div style="margin-bottom: 8px;"><strong>Definition:</strong> ${chosenClueEN}</div>
        <div style="font-size: clamp(0.9rem, 2.5vw, 1rem); color: #7c6485; border-top: 1px dashed #ffd1dc; padding-top: 8px;"><strong>意味:</strong> ${chosenClueJA}</div>
      `;
      document.getElementById("celebration-modal-overlay").classList.add("active");
    }

    function closeCelebrationModal() {
      document.getElementById("celebration-modal-overlay").classList.remove("active");
      nextWord();
    }

      if (letter >= 'A' && letter <= 'Z' && letter.length === 1) {
        const buttons = document.querySelectorAll(".letters button");
        for (let btn of buttons) {
          if (btn.textContent === letter && !btn.disabled) {
            btn.click();
            break;
          }
        }
      }
    });
