    <script>
        /**
         * MCQ DATA - Balanced for mobile
         * Rhythm: Easy, Easy, Difficult
         */
        const MCQS = [
            { q: "Where were the League's headquarters located?", a: "Geneva, Switzerland", o: ["Geneva, Switzerland", "Paris, France", "London, UK", "New York, USA"], d: "easy", ex: "The League chose Switzerland because it was a neutral nation." },
            { q: "Which body met once a year to debate policy?", a: "The Assembly", o: ["The Assembly", "The Council", "The Secretariat", "The ILO"], d: "easy", ex: "The Assembly allowed all member nations to have an equal vote." },
            { q: "What caused the US to never join the League?", a: "Senate Isolationism", o: ["Senate Isolationism", "Wilson's refusal", "British blocking", "Lack of funding"], d: "difficult", ex: "The US Senate refused to ratify the Treaty of Versailles." },
            { q: "What was the 'Collective Security' concept?", a: "All defend one", o: ["All defend one", "No nation fights", "USA pays for all", "Arms reduction"], d: "easy", ex: "Article 10 required members to protect each other from aggression." },
            { q: "Which group was known as the 'Civil Service'?", a: "The Secretariat", o: ["The Secretariat", "The Council", "The Assembly", "The Mandates"], d: "easy", ex: "The Secretariat handled the daily administration and paperwork." },
            { q: "How did the Council differ from the Assembly?", a: "Permanent Members", o: ["Permanent Members", "Higher tax rate", "Independent army", "Veto for everyone"], d: "difficult", ex: "Britain, France, Italy, and Japan had permanent seats and vetoes." },
            { q: "Which success involved a vote over a border?", a: "Upper Silesia", o: ["Upper Silesia", "Vilna Dispute", "Corfu Incident", "The Ruhr Crisis"], d: "easy", ex: "A plebiscite was held in 1921 to divide the land between Germany and Poland." },
            { q: "What triggered the 1925 Bulgaria dispute?", a: "A shooting of a dog", o: ["A shooting of a dog", "A stolen battleship", "An oil pipeline", "A border bridge"], d: "easy", ex: "A Greek soldier was shot while retrieving his dog across the border." },
            { q: "Why was the 1923 Corfu incident a failure?", a: "Mussolini defied it", o: ["Mussolini defied it", "Greece had no money", "USA sent weapons", "The League agreed"], d: "difficult", ex: "The League bowed to Italy, a permanent member of the Council." }
        ];

        // Tetris Setup
        const COLS = 10;
        const ROWS = 20;
        let blockSize = 0;
        const canvas = document.getElementById('game-canvas');
        const ctx = canvas.getContext('2d');
        const COLORS = ['#ef4444', '#3b82f6', '#eab308', '#8b5cf6', '#10b981'];

        let state = {
            grid: Array.from({ length: ROWS }, () => Array(COLS).fill(0)),
            score: 0,
            lives: 5,
            timeLeft: 180,
            gameActive: false,
            paused: false,
            piece: null,
            moveCount: 0,
            qIndex: 0,
            dropInterval: 1000,
            lastDrop: 0
        };

        const SHAPES = [
            [[1, 1, 1, 1]], // I
            [[1, 1], [1, 1]], // O
            [[0, 1, 0], [1, 1, 1]], // T
            [[1, 1, 0], [0, 1, 1]], // S
            [[0, 1, 1], [1, 1, 0]]  // Z
        ];

        function resizeCanvas() {
            const h = window.innerHeight * 0.55;
            blockSize = Math.floor(h / ROWS);
            canvas.width = blockSize * COLS;
            canvas.height = blockSize * ROWS;
        }

        function createPiece() {
            const shape = SHAPES[Math.floor(Math.random() * SHAPES.length)];
            const color = COLORS[Math.floor(Math.random() * COLORS.length)];
            return {
                pos: { x: Math.floor(COLS / 2) - 1, y: 0 },
                matrix: shape,
                color: color
            };
        }

        function draw() {
            ctx.fillStyle = '#0f172a';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Grid
            state.grid.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value) drawBlock(x, y, value);
                });
            });

            // Active Piece
            if (state.piece) {
                state.piece.matrix.forEach((row, y) => {
                    row.forEach((val, x) => {
                        if (val) drawBlock(state.piece.pos.x + x, state.piece.pos.y + y, state.piece.color);
                    });
                });
            }
        }

        function drawBlock(x, y, color) {
            ctx.fillStyle = color;
            ctx.fillRect(x * blockSize, y * blockSize, blockSize, blockSize);
            ctx.strokeStyle = '#0f172a';
            ctx.strokeRect(x * blockSize, y * blockSize, blockSize, blockSize);
        }

        function collide() {
            const [m, o] = [state.piece.matrix, state.piece.pos];
            for (let y = 0; y < m.length; ++y) {
                for (let x = 0; x < m[y].length; ++x) {
                    if (m[y][x] !== 0 && (state.grid[y + o.y] && state.grid[y + o.y][x + o.x]) !== 0) {
                        return true;
                    }
                }
            }
            return false;
        }

        function merge() {
            state.piece.matrix.forEach((row, y) => {
                row.forEach((value, x) => {
                    if (value !== 0) state.grid[y + state.piece.pos.y][x + state.piece.pos.x] = state.piece.color;
                });
            });
            state.moveCount++;
            if (state.moveCount >= 2) triggerMCQ();
        }

        function rotate(matrix) {
            return matrix[0].map((_, i) => matrix.map(row => row[i]).reverse());
        }

        function drop() {
            if (!state.gameActive || state.paused) return;
            state.piece.pos.y++;
            if (collide()) {
                state.piece.pos.y--;
                merge();
                clearLines();
                resetPiece();
            }
            state.lastDrop = Date.now();
        }

        function resetPiece() {
            state.piece = createPiece();
            if (collide()) endGame("TOWER CRUMBLED", "The League's foundations collapsed under pressure.");
        }

        function clearLines() {
            outer: for (let y = state.grid.length - 1; y > 0; --y) {
                for (let x = 0; x < state.grid[y].length; ++x) {
                    if (state.grid[y][x] === 0) continue outer;
                }
                const row = state.grid.splice(y, 1)[0].fill(0);
                state.grid.unshift(row);
                state.score += 100;
                y++;
            }
            updateHUD();
        }

        function triggerMCQ() {
            state.paused = true;
            state.moveCount = 0;
            const modal = document.getElementById('mcq-modal');
            const q = MCQS[state.qIndex % MCQS.length];
            
            document.getElementById('question-text').innerText = q.q;
            document.getElementById('difficulty-tag').innerText = `Level: ${q.d}`;
            document.getElementById('difficulty-tag').className = `text-[10px] px-2 py-1 rounded font-bold uppercase ${q.d === 'easy' ? 'bg-green-500/20 text-green-400' : 'bg-red-500/20 text-red-400'}`;
            
            const container = document.getElementById('options-container');
            container.innerHTML = '';
            document.getElementById('feedback-area').classList.add('hidden');
            container.classList.remove('hidden');

            q.o.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = "option-btn w-full px-4 rounded-xl bg-slate-800 border border-slate-700 text-white font-bold active:scale-95 text-xs";
                btn.innerText = opt;
                btn.onclick = () => handleAnswer(opt, q);
                container.appendChild(btn);
            });

            modal.classList.remove('hidden');
        }

        function handleAnswer(choice, q) {
            const container = document.getElementById('options-container');
            const feedback = document.getElementById('feedback-area');
            const fTitle = document.getElementById('feedback-title');
            const fText = document.getElementById('feedback-text');
            const nextBtn = document.getElementById('mcq-continue-btn');

            container.classList.add('hidden');
            feedback.classList.remove('hidden');

            if (choice === q.a) {
                state.score += 200;
                fTitle.innerText = "Correct!";
                fTitle.className = "font-black uppercase text-sm mb-1 text-green-500";
                fText.innerText = q.ex;
                nextBtn.innerText = "Resume Mission";
                nextBtn.onclick = () => {
                    state.qIndex++;
                    resume();
                };
            } else {
                state.lives--;
                fTitle.innerText = "Incorrect!";
                fTitle.className = "font-black uppercase text-sm mb-1 text-red-500";
                fText.innerText = `Correct Answer: ${q.a}\n\nReinforcement required. You must retry this fact.`;
                nextBtn.innerText = "Retry Question";
                nextBtn.onclick = triggerMCQ;
                updateHUD();
                if (state.lives <= 0) endGame("INTEGRITY FAILED", "Repeated historical errors led to a loss of international confidence.");
            }
        }

        function resume() {
            document.getElementById('mcq-modal').classList.add('hidden');
            state.paused = false;
        }

        function updateHUD() {
            document.getElementById('score').innerText = state.score.toString().padStart(4, '0');
            const m = Math.floor(state.timeLeft / 60);
            const s = state.timeLeft % 60;
            document.getElementById('timer').innerText = `${m}:${s < 10 ? '0' : ''}${s}`;
            document.getElementById('integrity-stars').innerHTML = '★'.repeat(state.lives) + '<span class="text-slate-800">' + '★'.repeat(5 - state.lives) + '</span>';
        }

        function endGame(status, msg) {
            state.gameActive = false;
            const screen = document.getElementById('end-screen');
            document.getElementById('end-status').innerText = status;
            document.getElementById('end-status').className = `text-4xl font-black mb-2 italic uppercase tracking-tighter ${status.includes('MASTER') ? 'text-green-500' : 'text-red-600'}`;
            document.getElementById('final-score').innerText = state.score;
            document.getElementById('end-message').innerText = msg;
            screen.classList.remove('hidden');
        }

        function startMission() {
            document.getElementById('start-screen').classList.add('hidden');
            state.gameActive = true;
            resetPiece();
            setInterval(() => {
                if (state.gameActive && !state.paused) {
                    state.timeLeft--;
                    updateHUD();
                    if (state.timeLeft <= 0) endGame("HISTORY MASTER", "You maintained League stability through the entire inter-war period!");
                }
            }, 1000);
            gameLoop();
        }

        function gameLoop() {
            if (!state.gameActive) return;
            const now = Date.now();
            if (now - state.lastDrop > state.dropInterval) drop();
            draw();
            requestAnimationFrame(gameLoop);
        }

        // Controls
        document.getElementById('ctrl-left').onclick = () => {
            if (state.paused) return;
            state.piece.pos.x--;
            if (collide()) state.piece.pos.x++;
        };
        document.getElementById('ctrl-right').onclick = () => {
            if (state.paused) return;
            state.piece.pos.x++;
            if (collide()) state.piece.pos.x--;
        };
        document.getElementById('ctrl-rotate').onclick = () => {
            if (state.paused) return;
            const m = state.piece.matrix;
            state.piece.matrix = rotate(m);
            if (collide()) state.piece.matrix = m;
        };
        document.getElementById('ctrl-drop').onclick = () => {
            if (state.paused) return;
            while (!collide()) state.piece.pos.y++;
            state.piece.pos.y--;
            merge();
            clearLines();
            resetPiece();
        };

        window.onload = () => {
            resizeCanvas();
            updateHUD();
            document.getElementById('start-btn').onclick = startMission;
        };
    </script>
