    <script>
        // Initialize Lucide Icons
        lucide.createIcons();

        // Filtering Logic
        function filterCards(filter) {
            const cards = document.querySelectorAll('.case-card');
            const buttons = document.querySelectorAll('.filter-btn');

            // Reset buttons styling
            buttons.forEach(btn => {
                btn.className = 'filter-btn px-4 py-2 rounded-full text-sm font-semibold bg-white text-slate-600 border border-slate-200 hover:bg-slate-50 transition cursor-pointer';
            });

            // Style active button
            const activeBtn = document.getElementById(`btn-${filter}`);
            activeBtn.className = 'filter-btn active px-4 py-2 rounded-full text-sm font-semibold bg-slate-800 text-white hover:bg-slate-700 transition cursor-pointer shadow-md';

            // Filter cards
            cards.forEach(card => {
                card.style.display = 'block'; // Reset to block to measure styling
                if (filter === 'all') {
                    card.classList.remove('hidden');
                    card.classList.add('card-enter');
                } else {
                    if (card.classList.contains(filter)) {
                        card.classList.remove('hidden');
                        card.classList.add('card-enter');
                    } else {
                        card.classList.add('hidden');
                        card.classList.remove('card-enter');
                    }
                }
            });
            
            // Re-trigger animation hack
            void document.getElementById('cards-container').offsetWidth;
        }
    </script>
