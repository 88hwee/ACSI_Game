<script>
const territoryData = [
    {name: "Alsace-Lorraine", lost: true, info: "Returned to France"},
    {name: "Polish Corridor", lost: true, info: "Given to Poland for sea access"},
    {name: "Saar Basin", lost: true, info: "Under League of Nations control for 15 years"},
    {name: "Rhineland", lost: false, info: "Demilitarized, not given away"},
    {name: "Danzig (Gda≈Ñsk)", lost: true, info: "Became a Free City under League control"},
    {name: "Sudetenland", lost: false, info: "This was taken later by Nazi Germany (1938)"},
    {name: "German Colonies", lost: true, info: "All overseas territories lost"},
    {name: "Bavaria", lost: false, info: "Remained part of Germany"},
    {name: "Northern Schleswig", lost: true, info: "Returned to Denmark after plebiscite"},
    {name: "Belgium", lost: false, info: "Was never German territory"},
    {name: "Eupen-Malm√©dy", lost: true, info: "Small region given to Belgium"},
    {name: "Memel Territory", lost: true, info: "Given to Lithuania"}
];

class TOVGame {
    constructor() {
        this.selectedTiles = new Set();
        this.gameChecked = false;
        this.init();
    }
    
    init() {
        this.createTiles();
        this.bindEvents();
        this.updateStats();
    }
    
    createTiles() {
        const tilesDiv = document.getElementById("tiles");
        tilesDiv.innerHTML = '';
        
        territoryData.forEach((territory, index) => {
            const tile = document.createElement("div");
            tile.className = "tile";
            tile.textContent = territory.name;
            tile.dataset.index = index;
            tile.title = territory.info;
            
            tile.addEventListener("click", () => this.toggleTile(tile, index));
            tilesDiv.appendChild(tile);
        });
    }
    
    toggleTile(tile, index) {
        if (this.gameChecked) return;
        
        tile.classList.toggle("selected");
        
        if (this.selectedTiles.has(index)) {
            this.selectedTiles.delete(index);
        } else {
            this.selectedTiles.add(index);
        }
        
        this.updateStats();
    }
    
    updateStats() {
        const selectedCount = this.selectedTiles.size;
        const totalLost = territoryData.filter(t => t.lost).length;
        
        document.getElementById("selectedCount").textContent = selectedCount;
        document.getElementById("totalLost").textContent = totalLost;
        
        // Update progress bar
        const progress = Math.min((selectedCount / totalLost) * 100, 100);
        document.getElementById("progressFill").style.width = `${progress}%`;
    }
    
    checkAnswer() {
        if (this.gameChecked) return;
        
        this.gameChecked = true;
        const tiles = document.querySelectorAll(".tile");
        let correct = 0;
        let incorrect = 0;
        let missed = 0;
        
        tiles.forEach((tile, index) => {
            const isSelected = this.selectedTiles.has(index);
            const shouldBeSelected = territoryData[index].lost;
            
            tile.classList.remove("selected");
            
            if (isSelected && shouldBeSelected) {
                tile.classList.add("correct");
                correct++;
            } else if (isSelected && !shouldBeSelected) {
                tile.classList.add("incorrect");
                incorrect++;
            } else if (!isSelected && shouldBeSelected) {
                tile.classList.add("missed");
                missed++;
            }
        });
        
        const totalLost = territoryData.filter(t => t.lost).length;
        const accuracy = Math.round((correct / totalLost) * 100);
        
        document.getElementById("accuracy").textContent = `${accuracy}%`;
        
        this.showResult(correct, incorrect, missed, totalLost, accuracy);
    }
    
    showResult(correct, incorrect, missed, totalLost, accuracy) {
        const resultDiv = document.getElementById("result");
        resultDiv.classList.add("show");
        
        let scoreClass = "score-poor";
        let message = "Keep studying! üìö";
        
        if (accuracy >= 90) {
            scoreClass = "score-excellent";
            message = "Excellent! You're a history expert! üèÜ";
        } else if (accuracy >= 70) {
            scoreClass = "score-good";
            message = "Good job! Room for improvement! üëç";
        }
        
        resultDiv.innerHTML = `
            <div class="${scoreClass}">
                <strong>${message}</strong><br>
                ‚úÖ Correct: <strong>${correct}</strong>/${totalLost} | 
                ‚ùå Wrong: <strong>${incorrect}</strong> | 
                ‚ö†Ô∏è Missed: <strong>${missed}</strong><br>
                <strong>Final Score: ${accuracy}%</strong>
            </div>
        `;
    }
    
    resetGame() {
        this.selectedTiles.clear();
        this.gameChecked = false;
        
        const tiles = document.querySelectorAll(".tile");
        tiles.forEach(tile => {
            tile.className = "tile";
        });
        
        document.getElementById("result").classList.remove("show");
        document.getElementById("result").innerHTML = "";
        document.getElementById("accuracy").textContent = "-";
        
        this.updateStats();
    }
    
    bindEvents() {
        document.getElementById("checkBtn").addEventListener("click", () => this.checkAnswer());
        document.getElementById("resetBtn").addEventListener("click", () => this.resetGame());
    }
}

// Initialize the game
new TOVGame();
</script>
