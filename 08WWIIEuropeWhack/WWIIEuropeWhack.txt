let state = {
    score: 0,
    timer: 60,
    combo: 0,
    lives: 3,
    active: false,
    bossActive: false,
    user: null,
    lastSpawn: 0

const grid = document.getElementById('grid');

function initGrid() {
    grid.innerHTML = '';
    for (let i = 0; i < 9; i++) {
        const hole = document.createElement('div');
        hole.className = 'hole';
        hole.innerHTML = `<div class="target" id="target-${i}" onclick="whack(${i})"><span class="target-text"></span></div>`;
        grid.appendChild(hole);
    }
}

async function initTracking() {
    const statsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    if (typeof __initial_auth_token !== 'undefined') {
        await signInWithCustomToken(auth, __initial_auth_token).catch(() => signInAnonymously(auth));
    } else {
        await signInAnonymously(auth);
    }
    
    onAuthStateChanged(auth, (user) => {
        state.user = user;
        if (user) {
            onSnapshot(statsDoc, (snap) => {
                if (snap.exists()) {
                    document.getElementById('global-play-count').innerText = (snap.data().playCount || 0).toLocaleString();
                    document.getElementById('play-tracker').style.opacity = '1';
                } else {
                    setDoc(statsDoc, { playCount: 0 });
                }
            });
        }
    });
}

async function recordPlay() {
    if (!state.user) return;
    const statsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    try { await updateDoc(statsDoc, { playCount: increment(1) }); }
    catch { await setDoc(statsDoc, { playCount: 1 }, { merge: true }); }
}

window.whack = (idx) => {
    if (!state.active || state.bossActive) return;
    const target = document.getElementById(`target-${idx}`);
    if (!target.classList.contains('show')) return;

    const isAmbition = target.classList.contains('ambition');
    target.classList.remove('show');

    if (isAmbition) {
        state.score += 10;
        state.combo += 1;
        updateHUD();
        if (state.combo > 0 && state.combo % 5 === 0) {
            triggerBoss();
        }
    } else {
        state.lives--;
        state.combo = 0;
        updateHUD();
        if (state.lives <= 0) endGame("DEFEATED");
    }
};

function triggerBoss() {
    state.bossActive = true;
    const q = bossQuestions[Math.floor(Math.random() * bossQuestions.length)];
    const overlay = document.getElementById('quiz-overlay');
    const qText = document.getElementById('question-text');
    const container = document.getElementById('options-container');
    const feedback = document.getElementById('feedback-text');

    qText.innerText = q.q;
    container.innerHTML = '';
    feedback.innerText = '';
    
    q.o.sort(() => Math.random() - 0.5).forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn';
        btn.innerText = opt;
        btn.onclick = () => {
            const btns = Array.from(container.children);
            btns.forEach(b => b.disabled = true);
            if (opt === q.a) {
                btn.classList.add('correct');
                feedback.innerText = "ACCURATE INTELLIGENCE!";
                feedback.style.color = "#22c55e";
                state.score += 50;
                updateHUD();
                setTimeout(() => {
                    overlay.style.display = 'none';
                    state.bossActive = false;
                }, 1500);
            } else {
                btn.classList.add('wrong');
                btns.forEach(b => { if(b.innerText === q.a) b.classList.add('correct'); });
                feedback.innerText = "INTEL COMPROMISED";
                feedback.style.color = "#ef4444";
                state.lives--;
                updateHUD();
                setTimeout(() => {
                    overlay.style.display = 'none';
                    state.bossActive = false;
                    if (state.lives <= 0) endGame("DEFEATED");
                }, 2500);
            }
        };
        container.appendChild(btn);
    });

    overlay.style.display = 'flex';
}

function updateHUD() {
    document.getElementById('stat-score').innerText = state.score;
    document.getElementById('stat-timer').innerText = Math.ceil(state.timer);
    document.getElementById('stat-combo').innerText = state.combo;
    document.getElementById('stat-lives').innerText = state.lives;
}

function spawn() {
    if (!state.active || state.bossActive) return;
    const idx = Math.floor(Math.random() * 9);
    const target = document.getElementById(`target-${idx}`);
    if (target.classList.contains('show')) return;

    const isAmbition = Math.random() > 0.4;
    const list = isAmbition ? ambitions : weaknesses;
    const text = list[Math.floor(Math.random() * list.length)];

    target.className = `target show ${isAmbition ? 'ambition' : 'weakness'}`;
    target.querySelector('.target-text').innerText = text;

    setTimeout(() => {
        target.classList.remove('show');
    }, 1500 + Math.random() * 1000);
}

function gameLoop(now) {
    if (state.active && !state.bossActive) {
        state.timer -= 0.016;
        if (now - state.lastSpawn > 800) {
            spawn();
            state.lastSpawn = now;
        }
        updateHUD();
        if (state.timer <= 0) endGame("TIME EXPIRED");
    }
    requestAnimationFrame(gameLoop);
}

function endGame(reason) {
    state.active = false;
    alert(`${reason}! FINAL SCORE: ${state.score}`);
    location.reload();
}

document.getElementById('btn-start').onclick = () => {
    recordPlay();
    document.getElementById('start-screen').style.display = 'none';
    state.active = true;
    state.lastSpawn = performance.now();
};

initGrid();
initTracking();
requestAnimationFrame(gameLoop);
