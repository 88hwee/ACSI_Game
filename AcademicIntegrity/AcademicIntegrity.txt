import React, { useState, useEffect, useRef } from 'react';
import { 
  ShieldCheck, 
  Clock, 
  BookOpen, 
  FileText, 
  Plus, 
  Trash2, 
  AlertCircle, 
  CheckCircle2, 
  Activity, 
  History,
  Download,
  Link2,
  Image as ImageIcon
} from 'lucide-react';

const App = () => {
  // --- State Management ---
  const [activeTab, setActiveTab] = useState('dashboard');
  const [isSessionActive, setIsSessionActive] = useState(false);
  const [sessionTime, setSessionTime] = useState(0);
  const [sources, setSources] = useState([]);
  const [evidence, setEvidence] = useState([]);
  const [draftContent, setDraftContent] = useState('');
  const [events, setEvents] = useState([]); // Log of all significant actions
  const [typingStats, setTypingStats] = useState({ keystrokes: 0, pastes: 0 });
  
  const timerRef = useRef(null);

  // --- Session Logic ---
  const toggleSession = () => {
    if (!isSessionActive) {
      logEvent('Session Started', 'Work period initiated');
      setIsSessionActive(true);
    } else {
      logEvent('Session Paused', `Worked for ${formatTime(sessionTime)}`);
      setIsSessionActive(false);
    }
  };

  useEffect(() => {
    if (isSessionActive) {
      timerRef.current = setInterval(() => {
        setSessionTime(prev => prev + 1);
      }, 1000);
    } else {
      clearInterval(timerRef.current);
    }
    return () => clearInterval(timerRef.current);
  }, [isSessionActive]);

  const logEvent = (action, details) => {
    const newEvent = {
      id: Date.now(),
      timestamp: new Date().toLocaleTimeString(),
      date: new Date().toLocaleDateString(),
      action,
      details
    };
    setEvents(prev => [newEvent, ...prev]);
  };

  const formatTime = (seconds) => {
    const h = Math.floor(seconds / 3600);
    const m = Math.floor((seconds % 3600) / 60);
    const s = seconds % 60;
    return `${h > 0 ? h + 'h ' : ''}${m}m ${s}s`;
  };

  // --- Source Logic ---
  const addSource = (e) => {
    e.preventDefault();
    const url = e.target.url.value;
    const title = e.target.title.value;
    if (!url || !title) return;

    const newSource = {
      id: Date.now(),
      url,
      title,
      addedAt: new Date().toLocaleString(),
      timeEngaged: 0,
      isVerified: url.includes('.edu') || url.includes('.gov') || url.includes('doi.org')
    };
    setSources([...sources, newSource]);
    logEvent('Source Added', `Added research: ${title}`);
    e.target.reset();
  };

  // --- Drafting & Typing Logic ---
  const handleDraftChange = (e) => {
    const newValue = e.target.value;
    if (newValue.length > draftContent.length + 50) {
      setTypingStats(prev => ({ ...prev, pastes: prev.pastes + 1 }));
      logEvent('Large Paste Detected', `Significant text block added (${newValue.length - draftContent.length} chars)`);
    } else {
      setTypingStats(prev => ({ ...prev, keystrokes: prev.keystrokes + 1 }));
    }
    setDraftContent(newValue);
  };

  // --- Components ---
  const Dashboard = () => (
    <div className="space-y-6">
      <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
        <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100">
          <div className="flex items-center gap-3 mb-2 text-indigo-600">
            <Clock size={20} />
            <h3 className="font-semibold text-sm uppercase tracking-wider">Session Time</h3>
          </div>
          <p className="text-3xl font-bold text-slate-800">{formatTime(sessionTime)}</p>
          <button 
            onClick={toggleSession}
            className={`mt-4 w-full py-2 rounded-xl font-medium transition-all ${
              isSessionActive 
              ? 'bg-rose-50 text-rose-600 hover:bg-rose-100' 
              : 'bg-indigo-600 text-white hover:bg-indigo-700 shadow-md'
            }`}
          >
            {isSessionActive ? 'Pause Work' : 'Start Working'}
          </button>
        </div>

        <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100">
          <div className="flex items-center gap-3 mb-2 text-emerald-600">
            <Activity size={20} />
            <h3 className="font-semibold text-sm uppercase tracking-wider">Work Integrity</h3>
          </div>
          <p className="text-3xl font-bold text-slate-800">
            {typingStats.keystrokes > 0 ? Math.round((typingStats.keystrokes / (typingStats.keystrokes + typingStats.pastes)) * 100) : 0}%
          </p>
          <p className="text-xs text-slate-400 mt-1">Manual Typing Ratio</p>
        </div>

        <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100">
          <div className="flex items-center gap-3 mb-2 text-amber-600">
            <BookOpen size={20} />
            <h3 className="font-semibold text-sm uppercase tracking-wider">Research Depth</h3>
          </div>
          <p className="text-3xl font-bold text-slate-800">{sources.length}</p>
          <p className="text-xs text-slate-400 mt-1">Verified Sources Logged</p>
        </div>
      </div>

      <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100">
        <h3 className="text-lg font-bold text-slate-800 mb-4 flex items-center gap-2">
          <History size={20} className="text-indigo-600" />
          Recent Activity Ledger
        </h3>
        <div className="space-y-3 max-h-64 overflow-y-auto pr-2">
          {events.length === 0 && <p className="text-slate-400 text-sm italic">No activity logged yet...</p>}
          {events.map(event => (
            <div key={event.id} className="flex items-start gap-4 p-3 rounded-lg bg-slate-50 border border-slate-100">
              <span className="text-xs font-mono text-indigo-400 whitespace-nowrap">{event.timestamp}</span>
              <div>
                <p className="text-sm font-semibold text-slate-700">{event.action}</p>
                <p className="text-xs text-slate-500">{event.details}</p>
              </div>
            </div>
          ))}
        </div>
      </div>
    </div>
  );

  const SourcesView = () => (
    <div className="space-y-6">
      <div className="bg-white p-6 rounded-2xl shadow-sm border border-slate-100">
        <h3 className="text-lg font-bold text-slate-800 mb-4">Add Research Source</h3>
        <form onSubmit={addSource} className="grid grid-cols-1 md:grid-cols-2 gap-4">
          <input 
            name="title" 
            placeholder="Document Title or Author" 
            className="px-4 py-2 bg-slate-50 border border-slate-200 rounded-xl focus:outline-none focus:ring-2 focus:ring-indigo-500"
          />
          <input 
            name="url" 
            placeholder="URL (e.g., JSTOR, Google Scholar link)" 
            className="px-4 py-2 bg-slate-50 border border-slate-200 rounded-xl focus:outline-none focus:ring-2 focus:ring-indigo-500"
          />
          <button type="submit" className="md:col-span-2 bg-slate-800 text-white py-2 rounded-xl hover:bg-slate-900 transition-colors flex items-center justify-center gap-2">
            <Plus size={18} /> Add to Research Log
          </button>
        </form>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
        {sources.map(source => (
          <div key={source.id} className="bg-white p-4 rounded-xl shadow-sm border border-slate-100 flex justify-between items-start">
            <div className="flex gap-3">
              <div className={`mt-1 ${source.isVerified ? 'text-emerald-500' : 'text-slate-300'}`}>
                <ShieldCheck size={20} />
              </div>
              <div>
                <h4 className="font-bold text-slate-800 leading-tight">{source.title}</h4>
                <a href={source.url} target="_blank" rel="noreferrer" className="text-xs text-indigo-500 hover:underline flex items-center gap-1 mt-1">
                  <Link2 size={12} /> View Source
                </a>
                <p className="text-[10px] text-slate-400 mt-2 uppercase tracking-tighter">Logged: {source.addedAt}</p>
              </div>
            </div>
            <button 
              onClick={() => {
                setSources(sources.filter(s => s.id !== source.id));
                logEvent('Source Removed', `Deleted: ${source.title}`);
              }}
              className="text-slate-300 hover:text-rose-500 transition-colors"
            >
              <Trash2 size={18} />
            </button>
          </div>
        ))}
      </div>
    </div>
  );

  const EditorView = () => (
    <div className="h-full flex flex-col gap-4">
      <div className="bg-white p-4 rounded-2xl shadow-sm border border-slate-100 flex justify-between items-center">
        <div className="flex items-center gap-4">
          <div className="flex flex-col">
            <span className="text-xs font-bold text-slate-400 uppercase">Words</span>
            <span className="text-lg font-mono font-bold text-slate-700">
              {draftContent.trim() ? draftContent.trim().split(/\s+/).length : 0}
            </span>
          </div>
          <div className="h-8 w-px bg-slate-100 mx-2" />
          <div className="flex flex-col">
            <span className="text-xs font-bold text-slate-400 uppercase">Status</span>
            <span className="flex items-center gap-1 text-emerald-500 text-sm font-bold">
              <ShieldCheck size={14} /> Origin Active
            </span>
          </div>
        </div>
        <div className="text-xs text-slate-400 italic">
          Recording typing cadence for integrity proof...
        </div>
      </div>
      <textarea
        value={draftContent}
        onChange={handleDraftChange}
        placeholder="Start writing your project here. Every change and paste is logged to provide proof of original thought."
        className="flex-grow p-8 bg-white rounded-2xl shadow-sm border border-slate-100 focus:outline-none focus:ring-2 focus:ring-indigo-100 text-slate-700 leading-relaxed text-lg resize-none font-serif"
      />
    </div>
  );

  const IntegrityReport = () => (
    <div className="bg-white p-8 rounded-3xl shadow-xl max-w-4xl mx-auto border border-slate-100">
      <div className="flex justify-between items-start mb-8 pb-6 border-b border-slate-100">
        <div>
          <div className="flex items-center gap-2 text-indigo-600 mb-1 font-black text-2xl">
            <ShieldCheck size={32} />
            ORIGIN
          </div>
          <p className="text-slate-500 font-medium">Academic Integrity Audit Trail</p>
        </div>
        <div className="text-right">
          <p className="text-sm font-bold text-slate-800">{new Date().toLocaleDateString()}</p>
          <p className="text-xs text-slate-400 tracking-widest">ID: {Math.random().toString(36).substr(2, 9).toUpperCase()}</p>
        </div>
      </div>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-8 mb-8">
        <div>
          <h4 className="text-sm font-bold text-slate-400 uppercase tracking-widest mb-4">Effort Metrics</h4>
          <div className="space-y-4">
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Total Work Duration</span>
              <span className="font-bold text-slate-800">{formatTime(sessionTime)}</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Keystroke Engagement</span>
              <span className="font-bold text-slate-800">{typingStats.keystrokes} interactions</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Human Typing Logic</span>
              <span className={`font-bold ${typingStats.pastes > 5 ? 'text-amber-500' : 'text-emerald-500'}`}>
                {typingStats.pastes < 3 ? 'Verified High' : 'Valid'}
              </span>
            </div>
          </div>
        </div>
        <div>
          <h4 className="text-sm font-bold text-slate-400 uppercase tracking-widest mb-4">Source Integrity</h4>
          <div className="space-y-4">
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Verified Academic Links</span>
              <span className="font-bold text-slate-800">{sources.filter(s => s.isVerified).length}</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Total Reference Count</span>
              <span className="font-bold text-slate-800">{sources.length}</span>
            </div>
            <div className="flex justify-between items-center">
              <span className="text-slate-600">Hallucination Risk</span>
              <span className="font-bold text-emerald-500">Minimal</span>
            </div>
          </div>
        </div>
      </div>

      <div className="bg-slate-50 p-6 rounded-2xl mb-8">
        <h4 className="text-sm font-bold text-slate-400 uppercase tracking-widest mb-3">Verification Timeline</h4>
        <div className="space-y-2">
          {events.slice(0, 5).map(e => (
            <div key={e.id} className="text-xs text-slate-600 flex gap-2">
              <span className="font-mono text-indigo-400">{e.timestamp}</span>
              <span>{e.action}: {e.details}</span>
            </div>
          ))}
          {events.length > 5 && <p className="text-xs text-slate-400 italic">...and {events.length - 5} other verified events</p>}
        </div>
      </div>

      <div className="flex gap-4">
        <button className="flex-grow flex items-center justify-center gap-2 bg-indigo-600 text-white py-3 rounded-xl font-bold hover:bg-indigo-700 transition-all shadow-lg shadow-indigo-100">
          <Download size={18} /> Export Full Audit (PDF)
        </button>
        <button className="flex-grow flex items-center justify-center gap-2 bg-slate-100 text-slate-700 py-3 rounded-xl font-bold hover:bg-slate-200 transition-all">
          <CheckCircle2 size={18} /> Seal for Submission
        </button>
      </div>
    </div>
  );

  return (
    <div className="min-h-screen bg-slate-50 font-sans text-slate-900 flex flex-col md:flex-row">
      {/* Sidebar Navigation */}
      <nav className="w-full md:w-64 bg-white border-r border-slate-200 p-6 flex flex-col">
        <div className="flex items-center gap-2 text-indigo-600 mb-8 font-black text-2xl tracking-tighter">
          <ShieldCheck size={28} />
          ORIGIN
        </div>
        
        <div className="space-y-2 flex-grow">
          {[
            { id: 'dashboard', icon: <Activity size={18} />, label: 'Pulse' },
            { id: 'editor', icon: <FileText size={18} />, label: 'Drafting' },
            { id: 'sources', icon: <BookOpen size={18} />, label: 'Sources' },
            { id: 'report', icon: <ShieldCheck size={18} />, label: 'Integrity' },
          ].map(item => (
            <button
              key={item.id}
              onClick={() => setActiveTab(item.id)}
              className={`w-full flex items-center gap-3 px-4 py-3 rounded-xl font-semibold transition-all ${
                activeTab === item.id 
                ? 'bg-indigo-50 text-indigo-600 shadow-sm' 
                : 'text-slate-500 hover:bg-slate-50'
              }`}
            >
              {item.icon}
              {item.label}
            </button>
          ))}
        </div>

        <div className="mt-auto pt-6 border-t border-slate-100">
          <div className="bg-indigo-600 rounded-2xl p-4 text-white">
            <p className="text-xs font-bold opacity-75 uppercase mb-1">Status</p>
            <p className="text-sm font-bold flex items-center gap-2">
              <span className={`w-2 h-2 rounded-full bg-white ${isSessionActive ? 'animate-pulse' : 'opacity-50'}`} />
              {isSessionActive ? 'Actively Protected' : 'Ready to Work'}
            </p>
          </div>
        </div>
      </nav>

      {/* Main Content Area */}
      <main className="flex-grow p-4 md:p-8 overflow-y-auto">
        <header className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8 gap-4">
          <div>
            <h1 className="text-2xl font-bold text-slate-800 capitalize">{activeTab}</h1>
            <p className="text-slate-500 text-sm">Managing: Chemistry Honors Thesis 2024</p>
          </div>
          <div className="flex gap-2">
             <div className="flex items-center gap-2 px-3 py-1.5 bg-white rounded-full border border-slate-200 text-xs font-bold text-slate-500 shadow-sm">
                <div className="w-2 h-2 rounded-full bg-emerald-500" />
                GPS Sync: University Lab
             </div>
          </div>
        </header>

        {activeTab === 'dashboard' && <Dashboard />}
        {activeTab === 'sources' && <SourcesView />}
        {activeTab === 'editor' && <EditorView />}
        {activeTab === 'report' && <IntegrityReport />}
      </main>

      {/* Simple Status Notification */}
      {typingStats.pastes > 2 && (
        <div className="fixed bottom-4 right-4 bg-amber-50 border border-amber-200 p-4 rounded-xl shadow-lg flex gap-3 max-w-sm animate-in slide-in-from-bottom-5">
          <AlertCircle className="text-amber-500 shrink-0" size={20} />
          <div>
            <p className="text-xs font-bold text-amber-800">High Paste Volume Detected</p>
            <p className="text-[10px] text-amber-700 mt-0.5">Large text blocks affect the "Human Authored" score. Ensure all pastes are cited appropriately in the Evidence Vault.</p>
          </div>
        </div>
      )}
    </div>
  );
};

export default App;
