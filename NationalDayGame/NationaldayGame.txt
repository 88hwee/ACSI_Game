        // ----------------- WEBAUDIO RETRO SYNTHESIZER -----------------
        class RetroAudio {
            constructor() {
                this.ctx = null;
                this.muted = false;
                this.bgmTimer = null;
            }

            init() {
                if (this.ctx) return;
                const AudioContextClass = window.AudioContext || window.webkitAudioContext;
                if (AudioContextClass) {
                    this.ctx = new AudioContextClass();
                }
            }

            playTone(freq, type, duration, vol = 0.1) {
                if (this.muted || !this.ctx) return;
                try {
                    this.init();
                    if (this.ctx.state === 'suspended') {
                        this.ctx.resume();
                    }
                    const osc = this.ctx.createOscillator();
                    const gain = this.ctx.createGain();

                    osc.type = type;
                    osc.frequency.setValueAtTime(freq, this.ctx.currentTime);

                    gain.gain.setValueAtTime(vol, this.ctx.currentTime);
                    gain.gain.exponentialRampToValueAtTime(0.00001, this.ctx.currentTime + duration);

                    osc.connect(gain);
                    gain.connect(this.ctx.destination);

                    osc.start();
                    osc.stop(this.ctx.currentTime + duration);
                } catch (e) {
                    console.error("Audio error", e);
                }
            }

            playJump() {
                this.playTone(150, 'triangle', 0.1, 0.15);
                setTimeout(() => this.playTone(300, 'triangle', 0.15, 0.15), 50);
            }

            playHitBlock() {
                this.playTone(100, 'sawtooth', 0.08, 0.2);
                setTimeout(() => this.playTone(220, 'square', 0.1, 0.15), 40);
            }

            playCorrect() {

                setTimeout(() => this.playTone(440, 'square', 0.1, 0.15), 100);

                this.platforms.push({ x: -200, y: this.groundY, w: 15000, h: 200 });

            // Loop and Physics Update
            loop(timestamp) {
                if (!this.playing) return;

                this.updatePhysics();
                this.render();

                requestAnimationFrame((t) => this.loop(t));

            updatePhysics() {
                const p = this.player;

                // Move Inputs
                let moveForce = 0.55;
                let maxSpeed = 5.5;
                let friction = 0.85;

                if (this.keys.left) {
                    p.vx -= moveForce;
                    p.facing = 'left';
                    p.animFrame += 0.15;
                } else if (this.keys.right) {
                    p.vx += moveForce;
                    p.facing = 'right';
                    p.animFrame += 0.15;
                } else {
                    p.vx *= friction;
                    p.animFrame = 0;
                }

                // Gravity
                p.vy += 0.6;

                // Clamp Horiz Speed
                if (p.vx > maxSpeed) p.vx = maxSpeed;
                if (p.vx < -maxSpeed) p.vx = -maxSpeed;

                // Update position
                p.x += p.vx;
                p.y += p.vy;

                // Keep player within left boundary
                if (p.x < 10) {
                    p.x = 10;
                    p.vx = 0;
                }

                // ABYSS RESCUE SHIELD: If Mario drops past the bottom viewport, pop him back up!
                if (p.y > this.groundY + 150) {
                    p.x = Math.max(100, p.x - 200); // Safety setback
                    p.y = this.groundY - 120; // High safe fall
                    p.vx = 0;
                    p.vy = 0;
                }

                // Collision with all physical platforms
                p.grounded = false;
                this.platforms.forEach(plat => {
                    if (p.x + p.width > plat.x && p.x < plat.x + plat.w) {
                        if (p.y + p.height >= plat.y && p.y + p.height - p.vy <= plat.y + 12) {
                            p.y = plat.y - p.height;
                            p.vy = 0;
                            p.grounded = true;
                        }
                    }
                });

                // Jump trigger
                if (this.keys.space && p.grounded) {
                    p.vy = -12;
                    p.grounded = false;
                    audio.playJump();
                }

                // Camera follow player
                let targetCam = p.x - canvas.width * 0.35;
                this.cameraX += (targetCam - this.cameraX) * 0.1;
                if (this.cameraX < 0) this.cameraX = 0;

                // Spawn endless blocks forward if the player runs extremely far
                if (p.x > this.blocks[this.blocks.length - 10].x) {
                    this.extendLevel();
                }

                // Handle Question Block Hit Detection
                this.blocks.forEach(block => {
                    if (block.bounceDir !== 0) {
                        block.bounceOffset += block.bounceDir * 2;
                        if (block.bounceOffset <= -12) {
                            block.bounceDir = 1;
                        } else if (block.bounceOffset >= 0) {
                            block.bounceOffset = 0;
                            block.bounceDir = 0;
                        }
                    }

                    if (block.state !== 'spent') {
                        if (p.x + p.width - 4 > block.x && p.x + 4 < block.x + block.width) {
                            if (p.y <= block.y + block.height && p.y - p.vy >= block.y + block.height - 12) {
                                p.y = block.y + block.height;
                                p.vy = 2;

                                block.bounceOffset = 0;
                                block.bounceDir = -1;

                                audio.playHitBlock();
                                this.triggerQuestion(block);
                            }
                        }
                    }
                });
            },

            // Continuously extend the platform level and solid floor infinitely
            extendLevel() {
                const len = this.blocks.length;
                const lastBlockX = this.blocks[len - 1].x;

                // Extend the physical solid red floor as well to match!
                this.platforms.push({ x: lastBlockX - 500, y: this.groundY, w: 8000, h: 200 });

                for (let i = 0; i < 20; i++) {
                    const blockX = lastBlockX + (i + 1) * 180;
                    const blockY = this.groundY - 110 - (i % 2) * 50;
                    this.blocks.push({
                        id: len + i,
                        x: blockX,
                        y: blockY,
                        width: 42,
                        height: 42,
                        state: 'active',
                        bounceOffset: 0,
                        bounceDir: 0
                    });

                    if (i % 3 === 1) {
                        this.platforms.push({
                            x: blockX - 40,
                            y: this.groundY - 70,
                            w: 120,
                            h: 15
                        });
                    }
                }
            },

                this.activeBlock = block;

                remText.innerText = this.questions.length;

                    const prefix = ["A", "B", "C", "D"][index];
                    btn.innerHTML = `
                        <span class="bg-slate-950 text-yellow-400 w-8 h-8 rounded-lg flex items-center justify-center font-bold text-sm group-hover:bg-red-600 group-hover:text-white transition-colors">${prefix}</span>
                        <span class="text-xs md:text-sm font-semibold flex-grow text-slate-200">${choice}</span>
                    `;

                    btn.onclick = () => this.handleAnswer(index, btn);
                    choicesDiv.appendChild(btn);
                });

                screen.classList.remove("hidden");
                setTimeout(() => {
                    box.classList.remove("scale-95", "opacity-0");
                    box.classList.add("scale-100", "opacity-100");
                }, 50);

            // Handle choice submission
            handleAnswer(choiceIndex, btnElement) {
                const q = this.questions[this.currentQIdx];
                const box = document.getElementById("question-box");
                const feedbackText = document.getElementById("modal-feedback");
                const allButtons = document.querySelectorAll(".choice-btn");

                allButtons.forEach(btn => btn.disabled = true);

                if (choiceIndex === q.a) {
                    audio.playCorrect();
                    btnElement.classList.add("border-emerald-500", "bg-emerald-950/40");
                    feedbackText.innerText = "✓ CORRECT! SG BONUS GAINED!";
                    feedbackText.className = "h-10 border-t border-emerald-800 bg-emerald-950 text-emerald-400 flex items-center justify-center text-xs font-semibold uppercase tracking-wider";
                    
                    this.score += 200;
                    this.answeredCount++;
                    this.activeBlock.state = 'spent';

                    this.questions.splice(this.currentQIdx, 1);

                    if (this.questions.length === 0) {
                        setTimeout(() => this.endGame(true, "VICTORY! ALL QUESTS COMPLETED"), 1200);
                        return;
                    }

                    if (this.currentQIdx >= this.questions.length) {
                        this.currentQIdx = 0;
                    }

                    this.player.vy = -6;

                } else {
                    audio.playWrong();
                    btnElement.classList.add("border-red-500", "bg-red-950/40");
                    box.classList.add("shake-element");
                    feedbackText.innerText = "✗ INCORRECT! WILL RETURN 2 QUESTIONS LATER!";
                    feedbackText.className = "h-10 border-t border-red-800 bg-red-950 text-red-400 flex items-center justify-center text-xs font-semibold uppercase tracking-wider animate-pulse";

                    this.activeBlock.state = 'wrong';

                    const failedQuestion = this.questions.splice(this.currentQIdx, 1)[0];
                    failedQuestion.hasSeen = true;

                    const targetIdx = Math.min(this.questions.length, 2);
                    this.questions.splice(targetIdx, 0, failedQuestion);

                    setTimeout(() => {
                        box.classList.remove("shake-element");
                    }, 400);
                }

                setTimeout(() => {
                    box.classList.remove("scale-100", "opacity-100");
                    box.classList.add("scale-95", "opacity-0");
                    document.getElementById("question-screen").classList.add("hidden");
                    
                    this.playing = true;
                    this.updateHUD();
                    requestAnimationFrame((t) => this.loop(t));
                }, 1400);
            },

            // Draw Parallax city skyline & characters
            render() {
                ctx.clearRect(0, 0, canvas.width, canvas.height);

                // 1. SKY BACKGROUND
                const skyGrad = ctx.createLinearGradient(0, 0, 0, canvas.height);
                skyGrad.addColorStop(0, "#2a080c");
                skyGrad.addColorStop(1, "#5c1118");
                ctx.fillStyle = skyGrad;
                ctx.fillRect(0, 0, canvas.width, canvas.height);

                // Drifting clouds
                ctx.fillStyle = "rgba(255, 255, 255, 0.05)";
                for (let i = 0; i < 5; i++) {
                    const cloudX = ((i * 300 - this.cameraX * 0.1) % (canvas.width + 200)) - 100;
                    const cloudY = 60 + (i % 2) * 40;
                    ctx.beginPath();
                    ctx.arc(cloudX, cloudY, 30, 0, Math.PI * 2);
                    ctx.arc(cloudX + 25, cloudY - 10, 35, 0, Math.PI * 2);
                    ctx.arc(cloudX + 50, cloudY, 25, 0, Math.PI * 2);
                    ctx.fill();
                }

                // 2. PARALLAX SILHOUETTE SKYLINE
                ctx.fillStyle = "#1e0508";
                const mbsOffset = -this.cameraX * 0.15;
                
                for (let i = 0; i < 4; i++) {
                    const startX = (i * 350 + mbsOffset) % (canvas.width + 300) - 100;
                    ctx.fillRect(startX, 240, 30, 150);
                    ctx.fillRect(startX + 45, 240, 30, 150);
                    ctx.fillRect(startX + 90, 240, 30, 150);
                    ctx.beginPath();
                    ctx.ellipse(startX + 60, 240, 80, 12, 0, 0, Math.PI * 2);
                    ctx.fill();
                    
                    ctx.beginPath();
                    ctx.arc(startX + 220, 260, 45, 0, Math.PI * 2, true);
                    ctx.lineWidth = 4;
                    ctx.strokeStyle = "rgba(239, 51, 64, 0.15)";
                    ctx.stroke();
                }

                // 3. GROUND PLATFORMS & DECORATIONS
                ctx.save();
                ctx.translate(-this.cameraX, 0);

                this.platforms.forEach(plat => {
                    ctx.fillStyle = "#EF3340";
                    ctx.fillRect(plat.x, plat.y, plat.w, plat.h);
                    
                    ctx.fillStyle = "#FFFFFF";
                    ctx.fillRect(plat.x, plat.y, plat.w, 4);

                    ctx.fillStyle = "#a81620";
                    for (let xPos = plat.x; xPos < plat.x + plat.w; xPos += 40) {
                        ctx.fillRect(xPos, plat.y + 10, 2, plat.h);
                        ctx.fillRect(xPos + 20, plat.y + 25, 2, plat.h);
                    }
                });

                // 4. DRAW QUESTION BLOCKS (SG CREST THEMED)
                this.blocks.forEach(block => {
                    const blockY = block.y + block.bounceOffset;

                    if (block.state === 'spent') {

                        ctx.lineWidth = 4;
                        ctx.beginPath();
                        ctx.moveTo(block.x + 12, blockY + 22);
                        ctx.lineTo(block.x + 20, blockY + 30);
                        ctx.lineTo(block.x + 30, blockY + 12);
                        ctx.stroke();

                    } else if (block.state === 'wrong') {
                        const pulse = Math.sin(Date.now() * 0.01) * 2;
                        ctx.fillStyle = "#ef4444";
                        ctx.fillRect(block.x - pulse, blockY - pulse, block.width + pulse * 2, block.height + pulse * 2);
                        ctx.strokeStyle = "#991b1b";
                        ctx.lineWidth = 3;
                        ctx.strokeRect(block.x - pulse, blockY - pulse, block.width + pulse * 2, block.height + pulse * 2);

                        ctx.strokeStyle = "#ffffff";
                        ctx.lineWidth = 4;
                        ctx.beginPath();
                        ctx.moveTo(block.x + 13, blockY + 13);
                        ctx.lineTo(block.x + 29, blockY + 29);
                        ctx.moveTo(block.x + 29, blockY + 13);
                        ctx.lineTo(block.x + 13, blockY + 29);
                        ctx.stroke();

                    } else {
                        ctx.fillStyle = "#eab308";
                        ctx.fillRect(block.x - 2, blockY - 2, block.width + 4, block.height + 4);

                        ctx.fillStyle = "#EF3340";
                        ctx.fillRect(block.x, blockY, block.width, block.height / 2);
                        ctx.fillStyle = "#FFFFFF";
                        ctx.fillRect(block.x, blockY + block.height / 2, block.width, block.height / 2);

                        ctx.fillStyle = "#eab308";
                        ctx.beginPath();
                        ctx.arc(block.x + 16, blockY + 20, 7, 0, Math.PI * 2);
