    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, deleteDoc, doc, onSnapshot } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'jigsaw-edu-hub';

        let user = null;
        let puzzles = [];
        let activePuzzle = null;
        let difficulty = 30;
        let solvedCount = 0;
        let pieceWidth, pieceHeight;

        const initAuth = async () => {
            if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                await signInWithCustomToken(auth, __initial_auth_token);
            } else {
                await signInAnonymously(auth);
            }
        };
        initAuth();

        onAuthStateChanged(auth, (u) => {
            user = u;
            if (user) listenToPuzzles();
        });

        function listenToPuzzles() {
            const puzzlesRef = collection(db, 'artifacts', appId, 'public', 'data', 'puzzles');
            onSnapshot(puzzlesRef, (snap) => {
                puzzles = snap.docs.map(d => ({ id: d.id, ...d.data() }));
                renderGallery();
                renderTeacherList();
                document.getElementById('loading').classList.add('hidden');
            });
        }

        window.addPuzzleToCloud = async () => {
            if (!user) return;
            const title = document.getElementById('newTitle');
            const file = document.getElementById('newFile');
            const btn = document.getElementById('addBtn');

            if (!title.value || !file.files[0]) return showToast("Name and image required");
            if (puzzles.length >= 10) return showToast("10 puzzles max");

            btn.disabled = true;
            btn.innerText = "Uploading...";
            
            try {
                const base64 = await resizeAndEncode(file.files[0]);
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'puzzles'), {
                    title: title.value,
                    image: base64,
                    createdAt: Date.now()
                });
                title.value = '';
                file.value = '';
                showToast("Added to Gallery!");
            } catch (e) {
                showToast("Failed. Use a smaller file.");
            } finally {
                btn.disabled = false;
                btn.innerText = "Add to Class Gallery";
            }
        };

        window.deletePuzzle = async (id) => {
            if (confirm("Delete this puzzle?")) {
                await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'puzzles', id));
            }
        };

        function renderGallery() {
            const list = document.getElementById('puzzle-list');
            const empty = document.getElementById('empty-state');
            document.getElementById('view-gallery').classList.remove('hidden');

            if (puzzles.length === 0) {
                list.innerHTML = '';
                empty.classList.remove('hidden');
                return;
            }
            empty.classList.add('hidden');

            list.innerHTML = puzzles.map(p => `
                <div onclick="startPuzzle('${p.id}')" class="bg-white rounded-2xl p-2 shadow-sm border border-slate-100 hover:shadow-md transition-all active:scale-95 group">
                    <div class="aspect-square rounded-xl overflow-hidden bg-slate-50 mb-2">
                        <img src="${p.image}" class="w-full h-full object-cover">
                    </div>
                    <div class="px-1 flex justify-between items-center">
                        <p class="text-[10px] font-bold text-slate-700 truncate w-24">${p.title}</p>
                        ${localStorage.getItem('done_'+p.id) ? '<span>⭐</span>' : ''}
                    </div>
                </div>
            `).join('');
        }

        function renderTeacherList() {
            const list = document.getElementById('teacher-list');
            list.innerHTML = puzzles.map(p => `
                <div class="flex items-center justify-between bg-slate-50 p-2 rounded-xl border border-slate-100 mb-2">
                    <span class="text-xs font-bold text-slate-600 truncate w-32">${p.title}</span>
                    <button onclick="deletePuzzle('${p.id}')" class="text-[10px] font-black text-red-400">DEL</button>
                </div>
            `).join('');
        }

        window.startPuzzle = (id) => {
            activePuzzle = puzzles.find(p => p.id === id);
            showView('game');
            initGame();
        };

        function initGame() {
            const board = document.getElementById('puzzle-board');
            const container = document.getElementById('puzzle-board-container');
            board.innerHTML = '';
            solvedCount = 0;

            const img = new Image();
            img.src = activePuzzle.image;
            img.onload = () => {
                const aspect = img.width / img.height;
                let bWidth = container.clientWidth - 20;
                let bHeight = bWidth / aspect;
                
                if (bHeight > container.clientHeight - 20) {
                    bHeight = container.clientHeight - 20;
                    bWidth = bHeight * aspect;
                }

                board.style.width = bWidth + 'px';
                board.style.height = bHeight + 'px';

                const rows = Math.round(Math.sqrt(difficulty / aspect));
                const cols = Math.round(difficulty / rows);
                pieceWidth = bWidth / cols;
                pieceHeight = bHeight / rows;

                document.getElementById('game-title').innerText = activePuzzle.title;
                document.getElementById('progress-chip').innerText = `0 / ${rows * cols} Pieces Found`;

                board.style.backgroundImage = `url(${img.src})`;
                board.style.backgroundSize = 'cover';
                board.style.opacity = '1';

                for (let r = 0; r < rows; r++) {
                    for (let c = 0; c < cols; c++) {
                        const p = document.createElement('div');
                        p.className = 'puzzle-piece';
                        p.style.width = pieceWidth + 'px';
                        p.style.height = pieceHeight + 'px';
                        p.style.backgroundImage = `url(${img.src})`;
                        p.style.backgroundSize = `${bWidth}px ${bHeight}px`;
                        p.style.backgroundPosition = `-${c * pieceWidth}px -${r * pieceHeight}px`;
                        p.dataset.tx = c * pieceWidth;
                        p.dataset.ty = r * pieceHeight;
                        p.style.left = Math.random() * (bWidth - pieceWidth) + 'px';
                        p.style.top = Math.random() * (bHeight - pieceHeight) + 'px';

                        setupTouchDrag(p, board);
                        board.appendChild(p);
                    }
                }
                setTimeout(() => { board.style.transition = 'opacity 2s'; board.style.opacity = '0.08'; }, 1500);
            };
        }

        function setupTouchDrag(el, board) {
            let sx, sy, ox, oy;
            const start = (e) => {
                const t = e.type.includes('touch') ? e.touches[0] : e;
                sx = t.clientX; sy = t.clientY; ox = el.offsetLeft; oy = el.offsetTop;
                el.style.zIndex = 1000;
                document.addEventListener('mousemove', move, {passive: false});
                document.addEventListener('touchmove', move, {passive: false});
                document.addEventListener('mouseup', end);
                document.addEventListener('touchend', end);
            };
            const move = (e) => {
                if (e.cancelable) e.preventDefault();
                const t = e.type.includes('touch') ? e.touches[0] : e;
                el.style.left = Math.max(0, Math.min(ox + t.clientX - sx, board.clientWidth - pieceWidth)) + 'px';
                el.style.top = Math.max(0, Math.min(oy + t.clientY - sy, board.clientHeight - pieceHeight)) + 'px';
            };
            const end = () => {
                document.removeEventListener('mousemove', move); document.removeEventListener('touchmove', move);
                document.removeEventListener('mouseup', end); document.removeEventListener('touchend', end);
                const tx = parseFloat(el.dataset.tx), ty = parseFloat(el.dataset.ty);
                if (Math.sqrt((el.offsetLeft - tx)**2 + (el.offsetTop - ty)**2) < 20) {
                    el.style.left = tx + 'px'; el.style.top = ty + 'px'; el.classList.add('snapped');
                    solvedCount++;
                    const total = board.querySelectorAll('.puzzle-piece').length;
                    document.getElementById('progress-chip').innerText = `${solvedCount} / ${total} Pieces Found`;
                    if (solvedCount === total) {
                        localStorage.setItem('done_'+activePuzzle.id, 'true');
                        showToast("Puzzle Complete! ⭐");
                        setTimeout(() => showView('gallery'), 2000);
                    }
                } else { el.style.zIndex = 10; }
            };
            el.addEventListener('mousedown', start);
            el.addEventListener('touchstart', start, {passive: false});
        }

        window.showView = (v) => {
            document.getElementById('view-gallery').classList.toggle('hidden', v !== 'gallery');
            document.getElementById('view-game').classList.toggle('hidden', v !== 'game');
            document.getElementById('viewGalleryBtn').classList.toggle('hidden', v === 'gallery');
        };

        window.toggleTeacherMode = () => {
            const m = document.getElementById('teacher-modal');
            m.classList.toggle('hidden');
            m.style.display = m.classList.contains('hidden') ? 'none' : 'flex';
        };

        window.shufflePuzzle = () => initGame();
        window.changeDifficulty = (v) => { difficulty = parseInt(v); initGame(); };
        function showToast(m) {
            const t = document.getElementById('toast');
            t.innerText = m; t.style.display = 'block';
            setTimeout(() => t.style.display = 'none', 3000);
        }
        async function resizeAndEncode(file) {
            return new Promise((res) => {
                const reader = new FileReader();
                reader.onload = (e) => {
                    const img = new Image();
                    img.onload = () => {
                        const canvas = document.createElement('canvas');
                        let w =
