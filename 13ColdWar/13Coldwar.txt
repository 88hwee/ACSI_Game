let sessionDeck = [];
function shuffle(a) { return a.sort(() => Math.random() - 0.5); }

function buildSessionDeck() {
    let easy = shuffle([...QUEST_BANK.easy]);
    let mid = shuffle([...QUEST_BANK.medium]);
    let diff = shuffle([...QUEST_BANK.hard]);
    
    let deck = [];
    while(easy.length > 0 || mid.length > 0 || diff.length > 0) {
        if(easy.length > 0) deck.push(easy.pop());
        if(easy.length > 0) deck.push(easy.pop());
        if(mid.length > 0) deck.push(mid.pop());
        if(diff.length > 0) deck.push(diff.pop());
    }
    return deck;
}
sessionDeck = buildSessionDeck();

function resize() {
    state.viewportW = window.innerWidth;
    state.viewportH = window.innerHeight;
    canvas.width = state.viewportW;
    canvas.height = state.viewportH;
    player.x = state.viewportW / 2 - player.w / 2;
}

const touch = { left: false, right: false };
function init() {
    window.addEventListener('resize', resize);
    window.addEventListener('orientationchange', resize);
    resize();
    
    document.getElementById('btn-start').onclick = () => {
        document.getElementById('start-screen').classList.add('hidden');
        state.gameStarted = true; state.gameActive = true;
    };
    document.getElementById('btn-restart').onclick = () => location.reload();

    const setupTouch = (id, key) => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { e.preventDefault(); touch[key] = true; });
        el.addEventListener('touchend', (e) => { e.preventDefault(); touch[key] = false; });
    };
    setupTouch('ctrl-left', 'left');
    setupTouch('ctrl-right', 'right');
    document.getElementById('ctrl-shoot').addEventListener('touchstart', (e) => {
        e.preventDefault();
        if (state.gameActive && !state.paused) shoot();
    });

    requestAnimationFrame(update);
}

function shoot() {
    state.bullets.push({ x: player.x + player.w/2 - 4, y: state.viewportH - 180, r: 8, speed: 12 });
}

function spawnEnemy() {
    const r = 35;
    state.enemies.push({
        x: Math.random() * (state.viewportW - r*2) + r,
        y: -r,
        r: r,
        speed: 2.5 + Math.random() * 2,
        type: 'mcq'
    });
}

function createFireworks(x, y) {
    const colors = ['#3b82f6', '#fbbf24', '#ef4444', '#22c55e', '#ffffff'];
    for (let i = 0; i < 60; i++) {
        state.particles.push({
            x: x,
            y: y,
            vx: (Math.random() - 0.5) * 15,
            vy: (Math.random() - 0.5) * 15,
            life: 1.0,
            color: colors[Math.floor(Math.random() * colors.length)],
            size: Math.random() * 5 + 2
        });
    }
}

function triggerQuiz() {
    state.paused = true;
    const q = sessionDeck[0];
    const modal = document.getElementById('quiz-overlay');
    const container = document.getElementById('options-container');
    const feedback = document.getElementById('feedback');
    
    document.getElementById('question-text').innerText = q.q;
    document.getElementById('stats-q').innerText = sessionDeck.length;
    container.innerHTML = '';
    feedback.innerText = '';
    
    shuffle([...q.o]).forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn';
        btn.innerText = opt;
        btn.onclick = () => {
            Array.from(container.children).forEach(b => b.disabled = true);
            if (opt === q.a) {
                state.score += 50;
                document.getElementById('stats-coins').innerText = state.score;
                createFireworks(state.viewportW / 2, state.viewportH / 2);
                sessionDeck.shift();
                if (sessionDeck.length === 0) endGame();
                else resume();
            } else {
                btn.classList.add('wrong');
                feedback.innerText = `Incorrect. Intel: ${q.a}`;
                feedback.className = "mt-6 text-center font-black uppercase italic text-red-500 text-xl";
                
                const missed = sessionDeck.shift();
                // Reinforcement: Repeat missed question 2 positions later
                sessionDeck.splice(2, 0, missed);
                
                setTimeout(() => {
                    resume();
                }, 2000);
            }
        };
        container.appendChild(btn);
    });
    modal.classList.remove('hidden');
}

function resume() {
    document.getElementById('quiz-overlay').classList.add('hidden');
    state.paused = false;
}

function update() {
    if (!state.gameStarted) { requestAnimationFrame(update); return; }

    if (!state.paused) {
        if (touch.left && player.x > 0) player.x -= player.speed;
        if (touch.right && player.x < state.viewportW - player.w) player.x += player.speed;

        if (Date.now() - state.lastSpawn > 2200) {
            spawnEnemy();
            state.lastSpawn = Date.now();
        }

        for (let i = state.bullets.length - 1; i >= 0; i--) {
            const b = state.bullets[i];
            b.y -= b.speed;
            if (b.y < -20) state.bullets.splice(i, 1);
        }

        for (let i = state.enemies.length - 1; i >= 0; i--) {
            const en = state.enemies[i];
            en.y += en.speed;
            if (en.y > state.viewportH + en.r) {
                state.enemies.splice(i, 1);
                continue;
            }

            for (let j = state.bullets.length - 1; j >= 0; j--) {
                const b = state.bullets[j];
                const dist = Math.hypot(b.x - en.x, b.y - en.y);
                if (dist < en.r + 15) {
                    state.bullets.splice(j, 1);
                    state.enemies.splice(i, 1);
                    triggerQuiz();
                    break;
                }
            }
        }
    }

    for (let i = state.particles.length - 1; i >= 0; i--) {
        const p = state.particles[i];
        p.x += p.vx;
        p.y += p.vy;
        p.life -= 0.02;
        if (p.life <= 0) state.particles.splice(i, 1);
    }

    render();
    requestAnimationFrame(update);
}

function render() {
    ctx.clearRect(0, 0, state.viewportW, state.viewportH);
    
    ctx.fillStyle = "rgba(255,255,255,0.2)";
    for(let i=0; i<30; i++) {
        ctx.fillRect((Date.now()/60 + i*141) % state.viewportW, (i*179) % state.viewportH, 2, 2);
    }

    ctx.fillStyle = "#3b82f6";
    ctx.fillRect(player.x, state.viewportH - 160, player.w, 45);
    ctx.fillStyle = "#60a5fa";
    ctx.fillRect(player.x + (player.w * 0.25), state.viewportH - 185, player.w * 0.5, 25);

    state.enemies.forEach(en => {
        ctx.beginPath();
        ctx.arc(en.x, en.y, en.r, 0, Math.PI * 2);
        ctx.fillStyle = "#dc2626";
        ctx.fill();
        ctx.strokeStyle = "#ffffff";
        ctx.lineWidth = 3;
        ctx.stroke();
        ctx.fillStyle = "#ffffff";
        ctx.font = "900 24px Inter";
        ctx.textAlign = "center";
        ctx.fillText("?", en.x, en.y + 8);
    });

    ctx.fillStyle = "#fbbf24";
    state.bullets.forEach(b => {
        ctx.fillRect(b.x, b.y, 8, 22);
    });

    state.particles.forEach(p => {
        ctx.globalAlpha = p.life;
        ctx.fillStyle = p.color;
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
        ctx.fill();
    });
    ctx.globalAlpha = 1.0;
}

function endGame() {
    state.gameActive = false;
    document.getElementById('final-score').innerText = state.score;
    document.getElementById('end-screen').classList.remove('hidden');
}

setInterval(() => {
    if (state.gameActive && !state.paused && state.timer > 0) {
        state.timer--;
        document.getElementById('stats-timer').innerText = state.timer + "s";
        if (state.timer <= 0) endGame();
    }
}, 1000);
