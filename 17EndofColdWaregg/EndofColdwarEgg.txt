        // FISHER-YATES SHUFFLE FOR UNBIASED RANDOMIZATION
        function shuffleDeck(array) {
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        // INITIALIZE MISSION VARIABLES
        function startGameSession() {
            state.timeRemaining = state.totalGameTime;
            state.score = 0;
            state.multiplier = 1.0;
            state.currentLayer = 4;
            state.totalTaps = 0;
            state.questionsPresented = 0;
            state.questionsAnsweredCorrectly = 0;
            state.inventory = { diamond: 0, armour: 0, unicorn: 0, music: 0, money: 0 };
            state.missedQuestions = [];
            state.currentQuestion = null;
            
            // Shuffle whole deck so order is entirely randomized
            state.shuffledQuestions = shuffleDeck([...coldWarSyllabusQuestions]);
            state.currentQuestionIndex = 0;
            
            // Setup Visual Displays
            updateEggVisualColor(state.activeEggColor);
            resetEggLayerHp(4);
            resetEggPhysique();
            updateVaultUI();
            updateGoalProgressUI();
            
            dom.statScore.innerText = state.score;
            dom.statCombo.innerText = state.multiplier.toFixed(1) + "x";

            // Main 180s Game Clock loop
            if (state.timers.mainGame) clearInterval(state.timers.mainGame);
            state.timers.mainGame = setInterval(() => {
                state.timeRemaining--;
                
                dom.timeText.innerText = state.timeRemaining + "s";
                const percentage = (state.timeRemaining / state.totalGameTime) * 100;
                dom.timeBar.style.width = `${percentage}%`;
                
                if (state.timeRemaining <= 15) {
                    dom.timeBar.className = "h-full bg-red-600 transition-all";
                    dom.timeText.classList.add('animate-pulse', 'text-red-500');
                } else {
                    dom.timeBar.className = "h-full bg-gradient-to-r from-red-500 to-amber-400 transition-all";
                    dom.timeText.classList.remove('animate-pulse', 'text-red-500');
                }

                if (state.timeRemaining <= 0) {
                    terminateGameSession();
                }
            }, 1000);

            // Instantly load the first continuous question
            loadNextContinuousQuestion();

            dom.gameLiveAlert.innerText = "🚀 MCQ stream active. Hit your 30 questions target!";
        }

        // LOAD NEXT SYSTEMATIC RANDOM QUESTION
        function loadNextContinuousQuestion() {
            if (state.timeRemaining <= 0) return;

            // Recyclable deck mechanism: Reshuffle if player answers all 40 within 3 mins
            if (state.currentQuestionIndex >= state.shuffledQuestions.length) {
                state.shuffledQuestions = shuffleDeck([...coldWarSyllabusQuestions]);
                state.currentQuestionIndex = 0;
            }

            const questionData = state.shuffledQuestions[state.currentQuestionIndex];
            state.currentQuestion = questionData;
            state.questionsPresented++;

            updateGoalProgressUI();

            // Set MCQ Text content
            dom.questionTheme.innerText = `THEME: ${questionData.theme.toUpperCase()}`;
            dom.questionText.innerText = questionData.text;
            
            // Assemble choices
            const originalCorrectText = questionData.options[questionData.correctIndex];
            
            dom.optionsContainer.innerHTML = '';
            questionData.options.forEach((optText, index) => {
                const optLetter = String.fromCharCode(65 + index);
                const button = document.createElement('button');
                button.className = "option-btn w-full p-3 md:p-3.5 rounded-xl bg-slate-800 hover:bg-slate-700/80 text-left text-sm md:text-base font-bold border border-slate-700/60 transition-all active:scale-[0.98] flex items-center gap-3";
                button.innerHTML = `
                    <span class="w-6 h-6 rounded-full bg-slate-950/80 flex items-center justify-center text-slate-400 font-extrabold text-xs">${optLetter}</span>
                    <span class="option-text text-slate-200">${optText}</span>
                `;
                button.addEventListener('click', () => handleOptionSelection(button, optText === originalCorrectText, originalCorrectText));
                dom.optionsContainer.appendChild(button);
            });

            // Clean feedback
            dom.questionFeedback.classList.add('hidden');

            // Countdown timer reset (8s per question)
            state.questionSecondsLeft = 8.0;
            dom.questionTimeoutText.innerText = "8.0s";
            dom.questionTimerBar.style.width = "100%";
            dom.questionTimerBar.className = "h-full bg-emerald-500 w-full transition-all";

            if (state.questionTimer) clearInterval(state.questionTimer);
            
            const startTimestamp = Date.now();
            const duration = 8000;

            state.questionTimer = setInterval(() => {
                const elapsed = Date.now() - startTimestamp;
                const remaining = Math.max(0, duration - elapsed);
                const percent = (remaining / duration) * 100;
                
                dom.questionTimeoutText.innerText = (remaining / 1000).toFixed(1) + "s";
                dom.questionTimerBar.style.width = `${percent}%`;

                if (percent < 40) {
                    dom.questionTimerBar.className = "h-full bg-red-600 w-full transition-all";
                } else if (percent < 70) {
                    dom.questionTimerBar.className = "h-full bg-amber-500 w-full transition-all";
                }

                if (remaining <= 0) {
                    clearInterval(state.questionTimer);
                    handleQuestionTimeout(originalCorrectText);
                }
            }, 50);

            state.currentQuestionIndex++;
            dom.gameDynamicHint.innerHTML = "👉 Click an option above to lock in your revision points!";
        }

        // EVALUATE SELECTED MCQ OPTIONS
        function handleOptionSelection(selectedButton, isCorrect, correctText) {
            clearInterval(state.questionTimer);
            
            const allOptionBtns = dom.optionsContainer.querySelectorAll('button');
            allOptionBtns.forEach(btn => btn.disabled = true);

            if (isCorrect) {
                audio.playCorrect();
                selectedButton.classList.remove('bg-slate-800');
                selectedButton.classList.add('bg-emerald-950/80', 'border-emerald-500', 'text-emerald-300');
                
                state.questionsAnsweredCorrectly++;
                state.multiplier = Math.min(3.0, state.multiplier + 0.1); 
                dom.statCombo.innerText = state.multiplier.toFixed(1) + "x";

                // Bonus logic: correct answer cracks egg shell significantly!
                state.layerHp = Math.max(0, state.layerHp - 5);
                updateLayerProgressUI();
                progressiveCrackReveal();
                
                const earnedPoints = Math.round(150 * state.multiplier);
                state.score += earnedPoints;
                dom.statScore.innerText = state.score;

                dom.questionFeedback.className = "mt-4 p-3 bg-emerald-950/60 border border-emerald-900/40 rounded-xl text-center text-sm font-bold text-emerald-400 block z-10";
                dom.questionFeedback.innerText = `🎉 CORRECT! +${earnedPoints} Points! (Egg damaged)`;
                dom.gameLiveAlert.innerText = `🔥 Good! ${state.currentQuestion.theme} topic points secured.`;
                
                spawnTapParticle(`+${earnedPoints}`, "text-emerald-400 font-extrabold");

                if (state.layerHp <= 0) {
                    setTimeout(() => {
                        handleEggLayerShatter();
                    }, 400);
                }

            } else {
                audio.playIncorrect();
                selectedButton.classList.remove('bg-slate-800');
                selectedButton.classList.add('bg-red-950/80', 'border-red-500', 'text-red-300');

                // Highlight correct selection
                allOptionBtns.forEach(btn => {
                    if (btn.querySelector('.option-text').innerText === correctText) {
                        btn.classList.add('border-emerald-500/50', 'bg-slate-800/80');
                    }
                });

                state.multiplier = 1.0; 
                dom.statCombo.innerText = "1.0x";
                
                logMissedQuestion(state.currentQuestion);

                dom.questionFeedback.className = "mt-4 p-3 bg-red-950/60 border border-red-900/40 rounded-xl text-center text-sm font-bold text-red-400 block z-10";
                dom.questionFeedback.innerText = `❌ Incorrect. Correct: "${correctText}"`;
                dom.gameLiveAlert.innerText = `🛡️ Concept missed. Correct: "${correctText}"`;
                
                spawnTapParticle("Review!", "text-red-400 font-extrabold");
            }

            // Instantly queue next question after 1.2s delay for perfect paced speed learning
            setTimeout(() => {
                loadNextContinuousQuestion();
            }, 1200);
        }

        function progressiveCrackReveal() {
            const hpLeftPercent = (state.layerHp / state.maxLayerHp) * 100;
            if (hpLeftPercent < 80) dom.crackPaths[0].classList.remove('opacity-0');
            if (hpLeftPercent < 60) dom.crackPaths[1].classList.remove('opacity-0');
            if (hpLeftPercent < 40) dom.crackPaths[2].classList.remove('opacity-0');
            if (hpLeftPercent < 20) {
                dom.crackPaths[3].classList.remove('opacity-0');
                dom.crackPaths[4].classList.remove('opacity-0');
            }
        }

        function handleEggLayerShatter() {
            audio.playCrack();
            spawnShatterParticles();

            state.currentLayer--;

            if (state.currentLayer > 0) {
                dom.eggLeft.classList.add('brightness-150');
                dom.eggRight.classList.add('brightness-150');
                setTimeout(() => {
                    dom.eggLeft.classList.remove('brightness-150');
                    dom.eggRight.classList.remove('brightness-150');
                }, 150);

                resetEggLayerHp(state.currentLayer);
                progressiveCrackReveal();
                dom.gameLiveAlert.innerText = `💥 Cracked! Layer ${state.currentLayer + 1} dismantled!`;
            } else {
                revealTreasureCore();
            }
        }

            dom.gameLiveAlert.innerText = `👑 Vault Unlocked: ${treasure.title}!`;

        function updateVaultUI() {
            let uniqueFound = 0;
            Object.keys(dom.vaultSlots).forEach(type => {
                const count = state.inventory[type] || 0;
                const slot = dom.vaultSlots[type];
                const badge = slot.querySelector('.count-badge');

                if (count > 0) {
                    slot.classList.remove('opacity-30');
                    slot.classList.add('opacity-100', 'border-amber-500/80', 'bg-slate-900');
                    badge.classList.remove('hidden');
                    badge.innerText = count;
                    uniqueFound++;
                } else {
                    slot.classList.add('opacity-30');
                    slot.classList.remove('opacity-100', 'border-amber-500/80', 'bg-slate-900');
                    badge.classList.add('hidden');
                }
            });
            dom.vaultCount.innerText = `${uniqueFound} / 5 types`;
        }

        function spawnShatterParticles() {
            for (let i = 0; i < 20; i++) {
                const frag = document.createElement('div');
                frag.className = "absolute w-4 h-4 rounded-md pointer-events-none z-30 transition-all duration-700 ease-out";
                const paletteColors = colorPalettes[state.activeEggColor].colors;
                frag.style.backgroundColor = paletteColors[Math.floor(Math.random() * paletteColors.length)];
                frag.style.left = `calc(50% + ${Math.floor(Math.random() * 40) - 20}px)`;
                frag.style.top = `calc(50% + ${Math.floor(Math.random() * 40) - 20}px)`;
                
                dom.floatingLayer.appendChild(frag);

                const destX = (Math.random() * 200 - 100);
                const destY = (Math.random() * -200 - 30);
                const rotate = Math.random() * 360;

                setTimeout(() => {
                    frag.style.transform = `translate(${destX}px, ${destY}px) rotate(${rotate}deg) scale(0.2)`;
                    frag.style.opacity = "0";
                }, 10);

                setTimeout(() => { frag.remove(); }, 800);
            }
        }

        function spawnTapParticle(text, customClasses = "", x, y) {
            const container = document.createElement('div');
            container.className = `absolute pointer-events-none z-30 font-extrabold text-xs md:text-sm particle ${customClasses}`;
            container.innerText = text;
            
            if (x !== undefined && y !== undefined) {
                container.style.left = `${x}px`;
                container.style.top = `${y}px`;
            } else {
                container.style.left = `calc(50% + ${Math.floor(Math.random() * 80) - 40}px)`;
                container.style.top = `40%`;
            }

            dom.floatingLayer.appendChild(container);
            setTimeout(() => { container.remove(); }, 800);
        }

        // TERMINATE GAME SESSION & GENERATE DETAILED REPORT
        function terminateGameSession() {
            clearInterval(state.timers.mainGame);
            if (state.questionTimer) clearInterval(state.questionTimer);

            audio.playShatter();

            const completedAmount = state.questionsPresented - 1; // subtract active one
            const correctAmount = state.questionsAnsweredCorrectly;
            
            dom.finalTotalAnswered.innerText = completedAmount;
            dom.finalQuestionsAnswered.innerText = `${correctAmount} / ${completedAmount}`;
            dom.finalScore.innerText = state.score;

            // Determine if they met the 30-questions practice threshold!
            const metThreshold = completedAmount >= state.targetQuestionsGoal;

            if (metThreshold) {
                dom.completionBadge.className = "inline-block px-3 py-1 bg-emerald-500/10 text-emerald-400 border border-emerald-500/20 rounded-full text-[10px] md:text-xs font-bold tracking-widest uppercase mb-3 animate-bounce";
                dom.completionBadge.innerText = "🎯 Goal Completed!";
                dom.finalTargetStatus.innerText = "PASSED";
                dom.finalTargetStatus.className = "text-xl md:text-2xl font-black text-emerald-400";
                dom.completionVerdict.innerHTML = `Excellent work! You answered <span class="text-emerald-400 font-bold">${completedAmount} questions</span> (target was 30), establishing a robust revision baseline for your Cold War studies!`;
            } else {
                dom.completionBadge.className = "inline-block px-3 py-1 bg-red-500/10 text-red-400 border border-red-500/20 rounded-full text-[10px] md:text-xs font-bold tracking-widest uppercase mb-3";
                dom.completionBadge.innerText = "⏳ Time Expired";
                dom.finalTargetStatus.innerText = "UNFINISHED";
                dom.finalTargetStatus.className = "text-sm font-black text-red-400";
                dom.completionVerdict.innerHTML = `You completed <span class="text-amber-400 font-bold">${completedAmount} questions</span>. Make sure to work slightly faster next time to reach the <span class="text-emerald-400 font-bold">30 questions minimum</span> target before the 3 minutes expire!`;
            }

            // Render Study Checklist
            dom.wrongAnswersContainer.innerHTML = '';
            if (state.missedQuestions.length === 0 && completedAmount > 0) {
                dom.wrongAnswersContainer.innerHTML = `
                    <div class="text-center py-4 text-emerald-400">
                        <span class="text-2xl block mb-2">⭐ HERO OF THE REVISION ⭐</span>
                        <p class="text-xs">Amazing accuracy! You got 100% correct during this session run.</p>
                    </div>
                `;
            } else if (completedAmount === 0) {
                dom.wrongAnswersContainer.innerHTML = `
                    <p class="text-xs text-slate-500 italic text-center py-4">No questions were attempted. Start over to test your skills!</p>
                `;
            } else {
                state.missedQuestions.forEach(q => {
                    const row = document.createElement('div');
                    row.className = "p-3 bg-slate-900 border-l-4 border-red-500 rounded-r-xl flex flex-col gap-1.5 text-xs";
                    row.innerHTML = `
                        <span class="text-[10px] font-extrabold text-slate-500 tracking-wider uppercase">${q.theme}</span>
                        <p class="font-bold text-slate-300">${q.text}</p>
                        <div class="text-emerald-400 font-semibold flex items-start gap-1">
                            <span class="text-xs font-bold">✓ Study Core:</span>
                            <span class="text-slate-300 font-medium">${q.reason}</span>
                        </div>
                    `;
                    dom.wrongAnswersContainer.appendChild(row);
                });
            }

            dom.gameScreen.classList.add('hidden');
            dom.gameoverScreen.classList.remove('hidden');
        }
