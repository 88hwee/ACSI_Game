function shuffle(arr) { 
    for (let i = arr.length - 1; i > 0; i--) { 
        const j = Math.floor(Math.random() * (i + 1)); 
        [arr[i], arr[j]] = [arr[j], arr[i]]; 
    } 
    return arr;
}

// Generate blocks across a long path
const blocks = Array.from({length: 45}, (_, i) => ({ 
    x: 500 + (i * 380), 
    y: 150 + Math.floor(Math.random() * 120), 
    w: 60, 
    h: 60, 
    active: true,
    boss: i >= 24 // Visually different for boss questions
}));

document.getElementById('btn-start').onclick = () => {
    document.getElementById('start-screen').style.display = 'none';
    gameStarted = true; gameActive = true;
    failSound.play().then(() => { failSound.pause(); failSound.currentTime = 0; }).catch(() => {});
};

const keys = {};
window.onkeydown = e => {
    if (e.code === 'Space' && gameActive && !isPaused) {
        if (player.jumpsAvailable > 0) {
            player.vy = player.jumpPower;
            player.jumpsAvailable--;
            player.grounded = false;
        }
    }
    keys[e.code] = true;
};
window.onkeyup = e => keys[e.code] = false;

function drawPlayer() {
    ctx.save();
    ctx.translate(player.x + player.w / 2, player.y + player.h / 2);
    if (!player.facingRight) ctx.scale(-1, 1);
    
    // Character - Simple Hero
    ctx.fillStyle = "#ffcc99"; // skin
    ctx.fillRect(-12, -28, 24, 24); // head
    ctx.fillStyle = "#3b82f6"; // shirt
    ctx.fillRect(-15, -4, 30, 26); // body
    ctx.fillStyle = "#1e40af"; // pants
    ctx.fillRect(-15, 22, 12, 12); 
    ctx.fillRect(3, 22, 12, 12);
    
    // Eyes
    ctx.fillStyle = "white";
    ctx.fillRect(2, -22, 6, 6);
    ctx.fillStyle = "black";
    ctx.fillRect(5, -20, 3, 3);
    
    ctx.restore();
}

function triggerQuiz() {
    isPaused = true;
    gameActive = false;
    
    const currentQ = questionsDeck[0];
    const questionEl = document.getElementById('question-text');
    const headerEl = document.getElementById('question-header');
    const feedbackEl = document.getElementById('feedback-area');
    const container = document.getElementById('options-container');
    
    headerEl.innerText = questionsAnswered >= 24 ? "BOSS STRENGTH CHALLENGE" : "STRENGTH IDENTIFICATION";
    headerEl.style.color = questionsAnswered >= 24 ? "#f87171" : "#4ade80";
    
    questionEl.innerText = currentQ.q;
    feedbackEl.innerText = "";
    container.innerHTML = '';
    
    // Shuffle the options within the question
    let opts = shuffle([...currentQ.a]); 

    opts.forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn'; 
        btn.innerText = opt;
        btn.onclick = () => {
            Array.from(container.children).forEach(b => b.disabled = true);
            
            if (opt === currentQ.correct) {
                btn.classList.add('correct');
                feedbackEl.style.color = "#22c55e";
                feedbackEl.innerText = "EXCELLENT!";
                questionsAnswered++;
                document.getElementById('stats-progress').innerText = `Quest: ${questionsAnswered}/34`;
                
                setTimeout(() => {
                    document.getElementById('quiz-overlay').style.display = 'none';
                    questionsDeck.shift(); 
                    if (questionsDeck.length === 0 || questionsAnswered >= 34) {
                        gameActive = false;
                        document.getElementById('message').style.display = 'block';
                        document.getElementById('message').innerText = "STRENGTH MASTERED!";
                    } else {
                        isPaused = false;
                        gameActive = true;
                    }
                }, 1000);
            } else {
                btn.classList.add('wrong');
                feedbackEl.style.color = "#ef4444";
                feedbackEl.innerText = `LEARN: The correct strength is ${currentQ.correct}`;
                
                setTimeout(() => {
                    document.getElementById('quiz-overlay').style.display = 'none';
                    isFalling = true;
                    failSound.play().catch(()=>{});
                    // Move current question to back if wrong, but we want chronology, 
                    // so we just leave it at the front for next block or next life
                    isPaused = false;
                }, 2500);
            }
        };
        container.appendChild(btn);
    });
    
    document.getElementById('quiz-overlay').style.display = 'flex';
}

function update() {
    if (!gameStarted) { requestAnimationFrame(update); return; }

    if (timer <= 0) {
        gameActive = false;
        document.getElementById('message').style.display = 'block';
        document.getElementById('message').innerText = "TIME UP!";
    }

    if (isFalling) {
        player.vy += gravity; 
        player.y += player.vy;
        if (player.y > canvas.height + 100) {
            lives--; 
            document.getElementById('stats-lives').innerText = `Lives: ${lives}`;
            if (lives <= 0) { 
                gameActive = false; 
                document.getElementById('message').style.display = 'block'; 
                document.getElementById('message').innerText = "TRY AGAIN"; 
                document.getElementById('message').style.color = "#ef4444";
            } else { 
                player.x = Math.max(100, player.x - 400); 
                player.y = 100; 
                player.vy = 0; 
                isFalling = false; 
                if (timer > 0) gameActive = true; 
            }
        }
    } else if (gameActive && !isPaused) {
        if (keys['ArrowRight'] || keys['KeyD']) { player.vx = player.speed; player.facingRight = true; }
        else if (keys['ArrowLeft'] || keys['KeyA']) { player.vx = -player.speed; player.facingRight = false; }
        else player.vx *= 0.8;

        player.vy += gravity; 
        player.x += player.vx; 
        player.y += player.vy;

        if (player.y + player.h > 450) { 
            player.y = 450 - player.h; 
            player.vy = 0; 
            player.grounded = true; 
            player.jumpsAvailable = 2; 
        }
        
        blocks.forEach((b) => {
            if (b.active && player.x < b.x + b.w && player.x + player.w > b.x && player.y < b.y + b.h && player.y + player.h > b.y) {
                if (player.vy < 0 && player.y > b.y + b.h - 20) { 
                    b.active = false; 
                    player.vy = 2; 
                    if(questionsDeck.length > 0) triggerQuiz(); 
                } 
                else if (player.vy > 0 && player.y + player.h < b.y + 20) {
                    player.y = b.y - player.h;
                    player.vy = 0;
                    player.grounded = true;
                    player.jumpsAvailable = 2;
                }
            }
        });

        if (player.x < 0) player.x = 0;
    }
    
    ctx.setTransform(1,0,0,1,0,0); 
    
    // Background
    const grd = ctx.createLinearGradient(0, 0, 0, canvas.height);
    grd.addColorStop(0, "#0ea5e9");
    grd.addColorStop(1, "#38bdf8");
    ctx.fillStyle = grd;
    ctx.fillRect(0,0,canvas.width,canvas.height);

    // Camera
    let camX = -player.x + 200; 
    if (camX > 0) camX = 0;
    ctx.translate(camX, 0);

    // Ground
    ctx.fillStyle = isFalling ? '#1e293b' : '#22c55e'; 
    ctx.fillRect(-1000, 450, 40000, 50);
    
    // Blocks
    blocks.forEach(b => { 
        if (b.active) { 
            ctx.fillStyle = b.boss ? '#ef4444' : '#fbbf24'; 
            ctx.fillRect(b.x, b.y, b.w, b.h); 
            ctx.strokeStyle = '#ffffff';
            ctx.lineWidth = 3;
            ctx.strokeRect(b.x+3, b.y+3, b.w-6, b.h-6);
            
            ctx.fillStyle = '#ffffff'; 
            ctx.font = 'bold 28px Arial'; 
            ctx.textAlign = 'center';
            ctx.fillText(b.boss ? '!' : '?', b.x + b.w/2, b.y + b.h/2 + 10); 
        } 
    });

    drawPlayer();
    requestAnimationFrame(update);
}

setInterval(() => { 
    if (gameActive && !isPaused && timer > 0) { 
        timer--; 
        document.getElementById('stats-timer').innerText = "Time: " + timer + "s"; 
    } 
}, 1000);

update();
</script>
