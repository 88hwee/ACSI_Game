<script type="module">
import { initializeApp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js';
import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js';
import { getFirestore, doc, getDoc, setDoc, updateDoc, increment, onSnapshot } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js';

// Configuration Safety
let auth, db, appId;
let firebaseEnabled = false;

try {
    if (typeof __firebase_config !== 'undefined' && __firebase_config) {
        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        auth = getAuth(app);
        db = getFirestore(app);
        appId = typeof __app_id !== 'undefined' ? __app_id : 'cu-chi-mastery-revised';
        firebaseEnabled = true;
    }
} catch (e) {
    console.warn("Firebase config missing or invalid.");
}

const canvas = document.getElementById('gameCanvas');
const ctx = canvas.getContext('2d');

let state = {
    gameStarted: false,
    gameActive: false,
    paused: false,
    inTunnel: false,
    timer: 180,
    intel: 0,
    viewportW: window.innerWidth,
    viewportH: window.innerHeight,
    user: null,
    scrollX: 0,
    groundGaps: [],
    intelBlocks: [],
    lastTime: performance.now()
};

const player = {
    x: 100, y: 0, w: 40, h: 60,
    vx: 0, vy: 0,
    speed: 7.5, jumpPower: -13.5,
    grounded: false,
    facingRight: true
};

const masterDeck = [
    { q: "What was the duration of the Korean War?", options: ["1945–1948", "1950–1953", "1954–1957", "1960–1963"], correct: "1950–1953" },
    { q: "Which battle led to the French withdrawal from Indochina in May 1954?", options: ["Dien Bien Phu", "Saigon", "Hue", "Ia Drang"], correct: "Dien Bien Phu" },
    { q: "What agreement partitioned Vietnam into North and South in July 1954?", options: ["Paris Accords", "Geneva Accords", "SEATO Treaty", "Tonkin Resolution"], correct: "Geneva Accords" },
    { q: "Along which parallel was Vietnam divided after the 1954 conference?", options: ["38th Parallel", "17th Parallel", "20th Parallel", "15th Parallel"], correct: "17th Parallel" },
    { q: "What programme relocated peasants into fortifications?", options: ["Search and Destroy", "Strategic Hamlet", "Containment", "Village Liberation"], correct: "Strategic Hamlet" },
    { q: "In what year was the Ho Chi Minh Trail established?", options: ["1954", "1956", "1959", "1965"], correct: "1959" },
    { q: "What was the formal name of the Viet Cong?", options: ["Viet Minh", "National Liberation Front", "Red Army", "People's Army"], correct: "National Liberation Front" },
    { q: "What does guerrilla mean in Spanish?", options: ["Hidden hunter", "Jungle trap", "Little war", "Sudden death"], correct: "Little war" },
    { q: "Approximately how long were the VC tunnel systems?", options: ["50km", "150km", "300km+", "1000km"], correct: "300km+" },
    { q: "What event in August 1964 increased US involvement?", options: ["Tet Offensive", "Gulf of Tonkin incident", "Dien Bien Phu", "Ia Drang"], correct: "Gulf of Tonkin incident" }
];

let questionsDeck = [...masterDeck].sort(() => Math.random() - 0.5);

function generateWorld() {
    state.groundGaps = [];
    state.intelBlocks = [];
    for (let i = 1; i < 60; i++) {
        // Random gaps in ground
        if (Math.random() > 0.75) {
            state.groundGaps.push({ x: i * 550, w: 140 });
        }
        // Intel blocks
        state.intelBlocks.push({
            x: i * 450 + Math.random() * 200,
            y: state.viewportH - 240 - Math.random() * 120,
            w: 55, h: 55, active: true
        });
    }
}

async function initTracking() {
    if (!firebaseEnabled) return;
    const statsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    
    try {
        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
            await signInWithCustomToken(auth, __initial_auth_token);
        } else {
            await signInAnonymously(auth);
        }
    } catch (err) { console.warn("Tracking auth failed."); }
    
    onAuthStateChanged(auth, (user) => {
        state.user = user;
        if (user) {
            onSnapshot(statsDoc, (snap) => {
                if (snap.exists()) {
                    document.getElementById('global-play-count').innerText = (snap.data().playCount || 0).toLocaleString();
                    document.getElementById('play-tracker').style.opacity = '1';
                } else {
                    setDoc(statsDoc, { playCount: 0 });
                }
            });
        }
    });
}

async function recordDeployment() {
    if (!firebaseEnabled || !state.user) return;
    const statsDoc = doc(db, 'artifacts', appId, 'public', 'data', 'usage', 'stats');
    try { await updateDoc(statsDoc, { playCount: increment(1) }); }
    catch { await setDoc(statsDoc, { playCount: 1 }, { merge: true }); }
}

const input = { left: false, right: false, jump: false };
function setupControls() {
    const bindTouch = (id, key) => {
        const el = document.getElementById(id);
        el.addEventListener('touchstart', (e) => { e.preventDefault(); input[key] = true; }, { passive: false });
        el.addEventListener('touchend', (e) => { e.preventDefault(); input[key] = false; }, { passive: false });
        el.addEventListener('touchcancel', (e) => { e.preventDefault(); input[key] = false; }, { passive: false });
    };
    bindTouch('btn-left', 'left');
    bindTouch('btn-right', 'right');
    bindTouch('btn-jump', 'jump');
    
    window.onkeydown = (e) => {
        if (e.code === 'ArrowLeft') input.left = true;
        if (e.code === 'ArrowRight') input.right = true;
        if (e.code === 'Space' || e.code === 'ArrowUp') input.jump = true;
    };
    window.onkeyup = (e) => {
        if (e.code === 'ArrowLeft') input.left = false;
        if (e.code === 'ArrowRight') input.right = false;
        if (e.code === 'Space' || e.code === 'ArrowUp') input.jump = false;
    };
}

function resize() {
    state.viewportW = window.innerWidth;
    state.viewportH = window.innerHeight;
    canvas.width = state.viewportW;
    canvas.height = state.viewportH;
    if (!state.gameStarted) generateWorld();
}

function triggerQuiz() {
    state.paused = true;
    const q = questionsDeck[0];
    const overlay = document.getElementById('quiz-overlay');
    const container = document.getElementById('options-container');
    const feedback = document.getElementById('quiz-feedback');
    
    document.getElementById('question-text').innerText = q.q;
    container.innerHTML = '';
    feedback.innerText = '';
    
    const shuffledOptions = [...q.options].sort(() => Math.random() - 0.5);
    
    shuffledOptions.forEach(opt => {
        const btn = document.createElement('button');
        btn.className = 'option-btn';
        btn.innerText = opt;
        btn.onclick = () => {
            const btns = container.querySelectorAll('button');
            btns.forEach(b => b.disabled = true);
            
            if (opt === q.correct) {
                btn.classList.add('correct');
                state.intel += 1;
                document.getElementById('stats-intel').innerText = `INTEL: ${state.intel}`;
                feedback.innerText = "Intel Decrypted Successfully";
                feedback.className = "mt-8 text-center font-black uppercase text-green-500 italic";
                questionsDeck.shift();
                if (questionsDeck.length === 0) questionsDeck = [...masterDeck].sort(() => Math.random() - 0.5);
                setTimeout(() => {
                    overlay.classList.add('hidden');
                    state.paused = false;
                }, 1200);
            } else {
                btn.classList.add('wrong');
                btns.forEach(b => { if(b.innerText === q.correct) b.classList.add('correct'); });
                feedback.innerText = `CORRECT INTEL: ${q.correct}`;
                feedback.className = "mt-8 text-center font-black uppercase text-red-500 italic";
                
                // Reinforcement Logic
                const missed = questionsDeck.shift();
                questionsDeck.splice(2, 0, missed);
                
                setTimeout(() => {
                    overlay.classList.add('hidden');
                    state.paused = false;
                }, 2800);
            }
        };
        container.appendChild(btn);
    });
    overlay.classList.remove('hidden');
}

function engineLoop(now) {
    const deltaTime = now - state.lastTime;
    state.lastTime = now;

    if (state.gameStarted && !state.paused) {
        update(deltaTime);
        render();
    }
    requestAnimationFrame(engineLoop);
}

function update(dt) {
    // Timer
    state.timer -= dt / 1000;
    document.getElementById('stats-timer').innerText = Math.max(0, Math.ceil(state.timer)) + "s";
    if (state.timer <= 0) {
        state.paused = true;
        alert("Mission Time Expired. Final Intel: " + state.intel);
        location.reload();
    }

    // Horizontal Movement
    if (input.left) { player.vx = -player.speed; player.facingRight = false; }
    else if (input.right) { player.vx = player.speed; player.facingRight = true; }
    else { player.vx *= 0.85; }

    // Jump
    if (input.jump && player.grounded) {
        player.vy = player.jumpPower;
        player.grounded = false;
    }

    // Physics
    player.vy += 0.65; // Gravity
    player.x += player.vx;
    player.y += player.vy;

    // Ground Logic
    const groundLevel = state.viewportH - 80;
    let overGap = false;
    state.groundGaps.forEach(gap => {
        if (player.x + player.w/2 > gap.x && player.x + player.w/2 < gap.x + gap.w) overGap = true;
    });

    if (!overGap && player.y + player.h > groundLevel) {
        player.y = groundLevel - player.h;
        player.vy = 0;
        player.grounded = true;
    } else if (overGap && player.y + player.h > groundLevel) {
        player.grounded = false;
    }

    // Transition to Tunnel
    if (player.y > state.viewportH + 50) {
        state.inTunnel = !state.inTunnel;
        player.y = -60;
        player.vx = 0;
        document.getElementById('stats-sector').innerText = state.inTunnel ? "Sector: Cu Chi Tunnels" : "Sector: Jungle Surface";
        document.getElementById('stats-sector').style.color = state.inTunnel ? "#ef4444" : "#60a5fa";
    }

    // Camera
    state.scrollX = -player.x + (state.viewportW * 0.2);

    // Intel Collisions
    state.intelBlocks.forEach(b => {
        if (b.active && player.x < b.x + b.w && player.x + player.w > b.x && player.y < b.y + b.h && player.y + player.h > b.y) {
            b.active = false;
            triggerQuiz();
        }
    });
}

function render() {
    ctx.setTransform(1, 0, 0, 1, 0, 0);
    ctx.clearRect(0, 0, state.viewportW, state.viewportH);
    
    // Sector Background
    const grad = ctx.createLinearGradient(0, 0, 0, state.viewportH);
    if (state.inTunnel) {
        grad.addColorStop(0, '#451a03'); grad.addColorStop(1, '#1c1917');
    } else {
        grad.addColorStop(0, '#064e3b'); grad.addColorStop(1, '#022c22');
    }
    ctx.fillStyle = grad;
    ctx.fillRect(0, 0, state.viewportW, state.viewportH);

    ctx.translate(state.scrollX, 0);

    // Draw Ground
    ctx.fillStyle = state.inTunnel ? '#78350f' : '#14532d';
    const groundRepeat = 2000;
    const startX = Math.floor(-state.scrollX / groundRepeat) * groundRepeat - groundRepeat;
    for(let i=0; i<6; i++) {
        ctx.fillRect(startX + i * groundRepeat, state.viewportH - 80, groundRepeat, 80);
    }
    
    // Draw Gaps
    ctx.fillStyle = '#020617';
    state.groundGaps.forEach(gap => {
        ctx.fillRect(gap.x, state.viewportH - 80, gap.w, 80);
        ctx.strokeStyle = 'rgba(255,255,255,0.15)';
        ctx.lineWidth = 2;
        ctx.beginPath();
        ctx.moveTo(gap.x, state.viewportH - 80); ctx.lineTo(gap.x, state.viewportH);
        ctx.moveTo(gap.x + gap.w, state.viewportH - 80); ctx.lineTo(gap.x + gap.w, state.viewportH);
        ctx.stroke();
    });

    // Intel Blocks
    state.intelBlocks.forEach(b => {
        if (!b.active) return;
        ctx.fillStyle = '#3b82f6';
        ctx.fillRect(b.x, b.y, b.w, b.h);
        ctx.strokeStyle = 'white';
        ctx.lineWidth = 3;
        ctx.strokeRect(b.x, b.y, b.w, b.h);
        ctx.fillStyle = 'white';
        ctx.font = '900 28px Inter';
        ctx.textAlign = 'center';
        ctx.fillText('?', b.x + b.w/2, b.y + b.h/2 + 10);
    });

    // Player (Tactical Icon)
    ctx.fillStyle = state.facingRight ? '#ef4444' : '#dc2626';
    ctx.fillRect(player.x, player.y, player.w, player.h);
    // Soldier Helmet
    ctx.fillStyle = '#064e3b';
    ctx.fillRect(player.x - 4, player.y, player.w + 8, 12);
    // Backpack
    ctx.fillStyle = '#1c1917';
    const backpackX = state.facingRight ? player.x - 8 : player.x + player.w;
    ctx.fillRect(backpackX, player.y + 15, 8, 30);
}

// Global Init
window.addEventListener('resize', resize);
resize();
setupControls();
initTracking();

document.getElementById('btn-start').onclick = () => {
    if (firebaseEnabled) recordDeployment();
    document.getElementById('start-screen').classList.add('hidden');
    state.gameStarted = true;
    state.lastTime = performance.now();
};

requestAnimationFrame(engineLoop);
</script>
