  <script>
    // Correct combinations for each row:
    // r1: Economic crisis (p_econ, e_unemployed, ex_econ)
    // r2: Backstairs intrigue (p_backstairs, e_vonpapen, ex_backstairs)
    // r3: Hitler/propaganda (p_hitler, e_propaganda, ex_hitler)
    const correctConfig = {
      r1: { point: "p_econ",      evidence: "e_unemployed", explanation: "ex_econ" },
      r2: { point: "p_backstairs", evidence: "e_vonpapen",   explanation: "ex_backstairs" },
      r3: { point: "p_hitler",    evidence: "e_propaganda", explanation: "ex_hitler" }
    };

    const feedbackEl = document.getElementById("feedback");
    const checkBtn = document.getElementById("check-btn");
    const resetBtn = document.getElementById("reset-btn");
    const toggleModel = document.getElementById("toggle-model");
    const modelAnswer = document.getElementById("model-answer");

    function clearFeedback() {
      feedbackEl.textContent = "";
      feedbackEl.classList.remove("good", "ok", "bad");
    }

    function setFeedback(msg, type) {
      feedbackEl.textContent = msg;
      feedbackEl.classList.remove("good", "ok", "bad");
      if (type) feedbackEl.classList.add(type);
    }

    function clearCellClasses() {
      document.querySelectorAll(".cell-point, .cell-evidence, .cell-explanation").forEach(cell => {
        cell.classList.remove("cell-correct", "cell-incorrect");
      });
    }

    function checkPEEL() {
      clearCellClasses();
      clearFeedback();

      const rows = document.querySelectorAll("tbody tr[data-row]");
      let fullCorrectRows = 0;
      let partiallyCorrectRows = 0;

      rows.forEach(row => {
        const rowId = row.dataset.row;
        const config = correctConfig[rowId];

        const pointCell = row.querySelector(".cell-point");
        const evCell = row.querySelector(".cell-evidence");
        const exCell = row.querySelector(".cell-explanation");

        const pointSelect = row.querySelector('select[data-role="point"]');
        const evSelect = row.querySelector('select[data-role="evidence"]');
        const exSelect = row.querySelector('select[data-role="explanation"]');

        const chosenPoint = pointSelect.value;
        const chosenEvidence = evSelect.value;
        const chosenExplanation = exSelect.value;

        let rowCorrectCount = 0;

        if (chosenPoint === config.point) {
          pointCell.classList.add("cell-correct");
          rowCorrectCount++;
        } else if (chosenPoint) {
          pointCell.classList.add("cell-incorrect");
        }

        if (chosenEvidence === config.evidence) {
          evCell.classList.add("cell-correct");
          rowCorrectCount++;
        } else if (chosenEvidence) {
          evCell.classList.add("cell-incorrect");
        }

        if (chosenExplanation === config.explanation) {
          exCell.classList.add("cell-correct");
          rowCorrectCount++;
        } else if (chosenExplanation) {
          exCell.classList.add("cell-incorrect");
        }

        if (rowCorrectCount === 3) {
          fullCorrectRows++;
        } else if (rowCorrectCount > 0) {
          partiallyCorrectRows++;
        }
      });

      if (fullCorrectRows === 3) {
        setFeedback("Excellent ‚Äì you‚Äôve built three strong PEEL paragraphs with well-matched points, evidence and explanation. üéâ", "good");
      } else if (fullCorrectRows >= 1 || partiallyCorrectRows >= 1) {
        setFeedback(`You have ${fullCorrectRows} paragraph(s) fully correct and ${partiallyCorrectRows} partially correct. Adjust the red cells ‚Äì think about which fact best proves each point.`, "ok");
      } else {
        setFeedback("Nothing matches well yet. Rethink which <Point> each <Evidence> and <Explanation> fits. Start by matching the Great Depression to economic crisis, and von Papen to backstairs intrigue.", "bad");
      }
    }

    function resetAll() {
      clearFeedback();
      clearCellClasses();
      document.querySelectorAll("select[data-role]").forEach(sel => {
        sel.value = "";
      });
      document.querySelectorAll("textarea[data-role='link']").forEach(t => {
        t.value = "";
      });
    }

    checkBtn.addEventListener("click", checkPEEL);
    resetBtn.addEventListener("click", resetAll);

    toggleModel.addEventListener("click", () => {
      const isVisible = modelAnswer.style.display === "block";
      modelAnswer.style.display = isVisible ? "none" : "block";
      toggleModel.textContent = isVisible ? "Show model PEEL ideas" : "Hide model PEEL ideas";
    });
  </script>
