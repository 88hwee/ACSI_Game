        let currentQuestionIndex = 0;
        let goalsScored = 0;
        let timeLeft = 60; // 60 seconds
        let timerInterval;
        let selectedAnswerIndex = null;
        let gameActive = false;

        const alphabet = ['A', 'B', 'C', 'D'];

        // DOM elements
        const startScreen = document.getElementById('start-screen');
        const quizScreen = document.getElementById('quiz-screen');
        const startButton = document.getElementById('start-button');
        const questionText = document.getElementById('question-text');
        const mcqOptions = document.getElementById('mcq-options');
        const timerDisplay = document.getElementById('timer-display');
        const scoreDisplay = document.getElementById('score-display');
        const qCountDisplay = document.getElementById('q-count-display');
        const resultMessageDiv = document.getElementById('result-message');
        const messageText = document.getElementById('message-text');
        const football = document.getElementById('football');

        function startChallenge() {
            // Hide start screen, show quiz screen
            startScreen.classList.add('hidden');
            quizScreen.classList.remove('hidden');
            
            // Reset state
            currentQuestionIndex = 0;
            goalsScored = 0;
            timeLeft = 60;
            gameActive = true;
            football.style.pointerEvents = 'auto'; // Re-enable ball

            loadQuestion();
            startTimer();
        }

        function startTimer() {
            timerDisplay.textContent = `Time Left: ${timeLeft}s`;
            timerInterval = setInterval(() => {
                if (timeLeft > 0) {
                    timeLeft--;
                    timerDisplay.textContent = `Time Left: ${timeLeft}s`;
                } else {
                    clearInterval(timerInterval);
                    endGame();
                }
            }, 1000);
        }

        function loadQuestion() {
            // End game if all questions are answered
            if (currentQuestionIndex >= quizData.length) {
                clearInterval(timerInterval);
                endGame(true); // Passed 'true' to indicate Qs completed
                return;
            }

            const currentQ = quizData[currentQuestionIndex];
            
            // Reset UI for new question
            selectedAnswerIndex = null;
            mcqOptions.innerHTML = '';
            resultMessageDiv.className = 'result-message info';
            messageText.textContent = "Select an answer and click the ball!";
            
            // Update question text and count
            questionText.textContent = currentQ.q;
            qCountDisplay.textContent = `Q: ${currentQuestionIndex + 1}/${quizData.length}`;

            // Create options
            currentQ.opts.forEach((option, index) => {
                const label = document.createElement('label');
                label.innerHTML = `
                    <input type="radio" name="answer" value="${index}" onclick="selectAnswer(${index})" class="mr-2">
                    <span class="font-medium">${alphabet[index]}) ${option}</span>
                `;
                mcqOptions.appendChild(label);
            });
        }
        
        function selectAnswer(index) {
            selectedAnswerIndex = index;
        }

        function kickBall() {
            if (!gameActive) return; // Ignore clicks if game is over
            
            // Check if an answer is selected
            if (selectedAnswerIndex === null) {
                messageText.innerHTML = "ü•Ö **FOUL!** Please select an answer option first!";
                resultMessageDiv.className = 'result-message error';
                football.style.transform = 'translateY(-10px)'; // A little "miss" animation
                setTimeout(() => football.style.transform = 'translateY(0)', 300);
                return;
            }
            
            // Disable further kicks until question transition is complete
            football.style.pointerEvents = 'none';

            const currentQ = quizData[currentQuestionIndex];
            const isCorrect = selectedAnswerIndex === currentQ.correct;

            // Simple animation/feedback: set transition for next transform
            football.style.transition = 'transform 0.5s ease-out';
            
            if (isCorrect) {
                // GOAL! The ball moves directly left towards the goal on the left side.
                goalsScored++;
                scoreDisplay.textContent = `Goals: ${goalsScored}`;
                messageText.innerHTML = `‚öΩ **GOAL!** ${currentQ.feedback}`;
                resultMessageDiv.className = 'result-message success';
                // Move towards the goal (left and slightly up)
                football.style.transform = 'translate(-160px, -20px) scale(0.5)'; 

            } else {
                // MISS! The ball moves in one of two random, missing directions.
                const correctOptLetter = alphabet[currentQ.correct];
                const correctOptText = currentQ.opts[currentQ.correct];
                messageText.innerHTML = `‚ùå **MISS!** Correct answer was **${correctOptLetter}) ${correctOptText}**.`;
                resultMessageDiv.className = 'result-message error';
                
                // Randomly select a 'miss' trajectory
                const missType = Math.random() > 0.5 ? 'high-right' : 'low-right';
                
                if (missType === 'high-right') {
                    // Fly high over the goal and to the right
                    football.style.transform = 'translate(100px, -100px)'; 
                } else {
                    // Miss wide to the right and low
                    football.style.transform = 'translate(150px, 30px)';
                }
            }

            // Move to next question after feedback
            setTimeout(() => {
                // Reset ball position for the next question
                football.style.transition = 'none';
                football.style.transform = 'translate(0, 0)';
                football.style.pointerEvents = 'auto'; // Re-enable ball
                
                currentQuestionIndex++;
                loadQuestion();

            }, 1800); // 1.8 seconds delay for animation/feedback
        }

        function endGame(allQsAnswered = false) {
            gameActive = false;
            clearInterval(timerInterval);
            football.style.pointerEvents = 'none'; // Disable kicking

            if (allQsAnswered) {
                messageText.innerHTML = `üèÜ **CHALLENGE COMPLETE!** You answered all 30 questions! Your final score: **${goalsScored} goals!**`;
            } else {
                messageText.innerHTML = `‚è≥ **TIME'S UP!** You scored **${goalsScored} goals** in 60 seconds. You attempted ${currentQuestionIndex} questions.`;
            }
            
            resultMessageDiv.className = 'result-message game-over';

            // Show reset button
            const resetButton = document.createElement('button');
            resetButton.textContent = "Play Again";
            resetButton.className = "bg-blue-600 hover:bg-blue-700 text-white font-bold py-2 px-4 rounded-full mt-4 transition duration-300";
            resetButton.onclick = () => location.reload(); // Simple reload to restart
            resultMessageDiv.appendChild(resetButton);
        }

        // Initialize the game listener
        startButton.addEventListener('click', startChallenge);

        // Load the first question display when the page loads (hidden initially)
        window.onload = function() {
            // Initialize elements that will be hidden
            loadQuestion();
            // Ensure quiz is hidden until start is clicked
            startScreen.classList.remove('hidden');
            quizScreen.classList.add('hidden');
        };
