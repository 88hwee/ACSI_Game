    <script>
        const masterQuestions = [
            { q: "Which nation first defied the League by invading Manchuria?", opts: ["Italy", "Germany", "Japan", "USSR"], a: "Japan" },
            { q: "What was the League's primary weakness?", opts: ["No US approval", "No standing army", "Too slow", "German control"], a: "No standing army" },
            { q: "What territory did Italy invade in 1935?", opts: ["Rhineland", "Spain", "Abyssinia", "Austria"], a: "Abyssinia" },
            { q: "Hitler primarily sought to overturn which treaty?", opts: ["Locarno", "Versailles", "Kellogg", "Brest-Litovsk"], a: "Versailles" },
            { q: "What territory did Hitler remilitarize in 1936?", opts: ["Sudetenland", "Poland", "Rhineland", "Austria"], a: "Rhineland" },
            { q: "Hitler's desire for 'living space' is called...", opts: ["Appeasement", "Autarky", "Lebensraum", "Revanchism"], a: "Lebensraum" },
            { q: "The union of Germany and Austria in 1938 was the...", opts: ["Anschluss", "Autonomy", "Sudetia", "Détente"], a: "Anschluss" },
            { q: "Which PM is associated with the policy of Appeasement?", opts: ["Churchill", "Eden", "Chamberlain", "Attlee"], a: "Chamberlain" },
            { q: "What region was ceded to Germany at the Munich Conference?", opts: ["Bohemia", "Sudetenland", "Danzig", "The Corridor"], a: "Sudetenland" },
            { q: "What action officially began WWII in Europe?", opts: ["Munich Pact", "Invasion of Poland", "Anschluss", "Rhineland"], a: "Invasion of Poland" },
            { q: "The 1939 Pact between Germany and USSR was...", opts: ["Molotov-Ribbentrop", "Anti-Comintern", "Steel Pact", "Axis Pact"], a: "Molotov-Ribbentrop" },
            { q: "A secret clause in the Nazi-Soviet Pact divided...", opts: ["France", "Poland", "Finland", "Czechoslovakia"], a: "Poland" },
            { q: "Why did Hitler sign the Pact with Stalin?", opts: ["Mutual respect", "To avoid a two-front war", "To spread Fascism", "Trade deal"], a: "To avoid a two-front war" },
            { q: "What area did Hitler seize in March 1939, breaking Munich?", opts: ["Rhineland", "Austria", "Rest of Czechoslovakia", "The Saar"], a: "Rest of Czechoslovakia" }
        ];

        let quizPool = [];
        let score = 0;
        let selectedAnswer = null;
        let isAiming = false;
        let aimAngle = 0;
        let ballX = 0;

        const ball = document.getElementById('ball');
        const keeper = document.getElementById('goalkeeper');
        const aimer = document.getElementById('aimer');
        const quizUI = document.getElementById('quiz-ui');
        const stadium = document.getElementById('stadium');
        const overlay = document.getElementById('overlay');
        const aimInstr = document.getElementById('aim-instructions');

        function shuffle(arr) {
            let array = [...arr];
            for (let i = array.length - 1; i > 0; i--) {
                const j = Math.floor(Math.random() * (i + 1));
                [array[i], array[j]] = [array[j], array[i]];
            }
            return array;
        }

        function initGame() {
            quizPool = shuffle(masterQuestions);
            score = 0;
            document.getElementById('score').innerText = "0";
            showNextQuestion();
        }

        function resetWholeGame() {
            overlay.style.display = 'none';
            document.getElementById('final-summary').classList.add('hidden');
            initGame();
        }

        function showNextQuestion() {
            if (quizPool.length === 0) {
                showFinal();
                return;
            }

            const q = quizPool[0];
            document.getElementById('q-count').innerText = quizPool.length;
            document.getElementById('question-text').innerText = q.q;
            
            const container = document.getElementById('options-container');
            container.innerHTML = '';
            
            // Randomize choice order every time a question is shown
            const randomizedOpts = shuffle(q.opts);
            
            randomizedOpts.forEach(opt => {
                const btn = document.createElement('button');
                btn.className = 'option-btn';
                btn.innerText = opt;
                btn.onclick = () => handleSelection(opt);
                container.appendChild(btn);
            });

            quizUI.style.display = 'block';
            quizUI.style.opacity = '1';
            aimInstr.style.display = 'none';
            aimer.style.display = 'none';
            isAiming = false;
        }

        function handleSelection(choice) {
            selectedAnswer = choice;
            quizUI.style.opacity = '0';
            setTimeout(() => {
                quizUI.style.display = 'none';
                startAiming();
            }, 300);
        }

        function startAiming() {
            isAiming = true;
            aimAngle = 0;
            ballX = 0;
            updateAimPos();
            aimer.style.display = 'block';
            aimInstr.style.display = 'block';
        }

        function updateAimPos() {
            ball.style.transform = `translateX(calc(-50% + ${ballX}px))`;
            aimer.style.left = `calc(50% + ${ballX}px)`;
            aimer.style.transform = `translateX(-50%) rotate(${aimAngle}deg)`;
        }

        window.addEventListener('keydown', (e) => {
            if (!isAiming) return;

            if (e.code === 'ArrowLeft') {
                aimAngle = Math.max(aimAngle - 4, -45);
                ballX = Math.max(ballX - 8, -120);
                updateAimPos();
            } else if (e.code === 'ArrowRight') {
                aimAngle = Math.min(aimAngle + 4, 45);
                ballX = Math.min(ballX + 8, 120);
                updateAimPos();
            } else if (e.code === 'Space') {
                takeShot();
            }
        });

        function takeShot() {
            isAiming = false;
            aimInstr.style.display = 'none';
            aimer.style.display = 'none';

            const currentQ = quizPool[0];
            const isCorrect = selectedAnswer === currentQ.a;
            const isMiracleSave = isCorrect && (Math.random() < 0.15); 
            
            const targetX = ballX + (aimAngle * 3); 
            const targetY = -480;

            if (isCorrect && !isMiracleSave) {
                ball.style.transform = `translate(${targetX}px, ${targetY}px) scale(0.3)`;
                const dive = targetX > 0 ? -150 : 150;
                keeper.style.transform = `translateX(${dive}px) translateY(30px) rotate(${dive > 0 ? 60 : -60}deg)`;
                
                setTimeout(() => {
                    score++;
                    document.getElementById('score').innerText = score;
                    showFeedback("GOAL", true, "INTEL ACCURATE!");
                    quizPool.shift(); 
                }, 600);
            } else if (isCorrect && isMiracleSave) {
                ball.style.transform = `translate(${targetX}px, ${targetY + 40}px) scale(0.5)`;
                keeper.style.transform = `translateX(${targetX - 40}px) scale(1.2)`;
                
                setTimeout(() => {
                    showFeedback("SAVED!", false, "MIRACLE SAVE! Accurate intel, but the enemy intercepted.");
                    quizPool.shift(); 
                }, 600);
            } else {
                ball.style.transform = `translate(${targetX}px, ${targetY + 40}px) scale(0.5)`;
                keeper.style.transform = `translateX(${targetX - 40}px) scale(1.2)`;
                
                setTimeout(() => {
                    showFeedback("SAVED!", false, `INTEL ERROR! CORRECT: ${currentQ.a}`);
                    const missed = quizPool.shift();
                    quizPool.push(missed);
                }, 600);
            }
        }

        function showFeedback(result, success, message) {
            overlay.style.display = 'flex';
            const resEl = document.getElementById('overlay-result');
            resEl.innerText = result;
            resEl.className = `text-9xl italic mb-4 ${success ? 'text-green-500' : 'text-red-600'}`;
            document.getElementById('overlay-feedback').innerText = message;
            stadium.className = `stadium ${success ? 'flash-goal' : 'flash-miss'}`;

            setTimeout(() => {
                overlay.style.display = 'none';
                stadium.className = 'stadium';
                resetBall();
                showNextQuestion();
            }, 2000);
        }

        function resetBall() {
            ball.style.transition = 'none';
            ball.style.transform = 'translateX(-50%)';
            keeper.style.transition = 'none';
            keeper.style.transform = 'translateX(-50%)';
            setTimeout(() => {
                ball.style.transition = 'all 0.6s cubic-bezier(0.25, 0.46, 0.45, 0.94)';
                keeper.style.transition = 'all 0.4s cubic-bezier(0.175, 0.885, 0.32, 1.275)';
            }, 50);
        }

        function showFinal() {
            overlay.style.display = 'flex';
            document.getElementById('overlay-result').innerText = "FINISH";
            document.getElementById('overlay-feedback').innerText = `TOTAL GOALS: ${score}`;
            document.getElementById('final-summary').classList.remove('hidden');
            document.getElementById('poem-container').innerHTML = `
                "First they came for the Communists, and I did not speak out— Because I was not a Communist...<br><br>
                Then they came for the Jews, and I did not speak out— Because I was not a Jew...<br><br>
                Then they came for me—and there was no one left to speak for me."<br>
                <small>— Martin Niemöller</small><br><br>
                <strong>Dietrich Bonhoeffer:</strong> Pastor and resistance fighter executed in 1945 for his stand against tyranny.
            `;
        }

        initGame();
    </script>
